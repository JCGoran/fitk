222class FisherPlotter:
223    """
224    Class for plotting FisherMatrix objects.
225
226    Examples
227    --------
228    Create a Fisher plotter:
229    >>> fm1 = FisherMatrix(np.diag([1, 2, 3]), names=('a', 'b', 'c'))
230    >>> fm2 = FisherMatrix(np.diag([4, 5, 6]), names=('a', 'b', 'c'))
231    >>> fp = FisherPlotter(fm1, fm2)
232
233    Make a marginalized 1D plot:
234    >>> plot1d = fp.plot_1d()
235    >>> plot1d # doctest: +SKIP
236    <FisherFigure(
237        names=array(['a', 'b', 'c'], dtype=object),
238        figure=<Figure size 600x200 with 3 Axes>,
239        axes=array([<AxesSubplot:xlabel='a', ylabel='$p (\\theta)$'>,
240           <AxesSubplot:xlabel='b'>, <AxesSubplot:xlabel='c'>], dtype=object))>
241
242    Make a triangle plot:
243    >>> plot2d = fp.plot_triangle()
244    >>> plot2d
245    <FisherFigure(
246        names=array(['a', 'b', 'c'], dtype=object),
247        figure=<Figure size 600x600 with 6 Axes>,
248        axes=array([[<AxesSubplot:>, <AxesSubplot:>, <AxesSubplot:>],
249           [<AxesSubplot:ylabel='b'>, <AxesSubplot:>, <AxesSubplot:>],
250           [<AxesSubplot:xlabel='a', ylabel='c'>, <AxesSubplot:xlabel='b'>,
251            <AxesSubplot:xlabel='c'>]], dtype=object))>
252
253    They can be saved easily:
254    >>> plot1d.savefig('example_plot_1d.pdf') # doctest: +SKIP
255    """
256
257    def __init__(
258        self,
259        *args: FisherMatrix,
260        labels: Optional[Collection[str]] = None,
261        ylabel1d: str = r"$p (\theta)$",
262    ):
263        """
264        Constructor.
265
266        Parameters
267        ----------
268        args : FisherMatrix
269            `FisherMatrix` objects which we want to plot.
270            Can have different fiducial values.
271            The order of plotting of the parameters and the LaTeX labels to use
272            are determined by the first argument.
273
274        labels : Optional[Collection[str]] = None
275            the list of labels to put in the legend of the plots.
276            If not set, defaults to `0, ..., len(args) - 1`
277
278        Raises
279        ------
280        * `MismatchingSizeError` if the sizes of the inputs are not equal
281        * `ValueError` if the names of the inputs do not match
282
283        Examples
284        --------
285        Create a Fisher plotter with two objects:
286        >>> fm1 = FisherMatrix(np.diag([1, 2, 3]), names=('a', 'b', 'c'))
287        >>> fm2 = FisherMatrix(np.diag([4, 5, 6]), names=('a', 'b', 'c'))
288        >>> fp = FisherPlotter(fm1, fm2)
289        """
290        # make sure all of the Fisher objects have the same sizes
291        if not all(len(args[0]) == len(arg) for arg in args):
292            raise MismatchingSizeError(*args)
293
294        # check names
295        if not all(set(args[0].names) == set(arg.names) for arg in args):
296            raise ValueError("The names of the inputs do not match")
297
298        if labels is not None:
299            if len(labels) != len(args):
300                raise MismatchingSizeError(labels, args)
301        # default labels are 0, ..., n - 1
302        else:
303            labels = list(map(str, np.arange(len(args))))
304
305        # in case the names are shuffled, we sort them according to the FIRST
306        # input
307        indices = np.array(
308            [get_index_of_other_array(args[0].names, arg.names) for arg in args],
309            dtype=object,
310        )
311
312        self._values = [arg.sort(key=index) for index, arg in zip(indices, args)]
313
314        self._labels = np.array(labels, dtype=object)
315
316        self._ylabel1d = ylabel1d
317

345    def plot_1d(
346        self,
347        scale: float = 1.0,
348        max_cols: Optional[int] = None,
349        mpl_options: Optional[dict] = None,
350        **kwargs,
351    ):
352        """
353        Makes a 1D plot (Gaussians) of the Fisher objects
354
355        Parameters
356        ----------
357        max_cols : Optional[int] = None
358            the maximum number of columns to force the plot into.
359            By default, the parameters are always plotted horizontally; if you
360            need to spread it over `max_cols`, pass a non-negative integer
361            here.
362
363        mpl_options : Optional[dict] = None
364            any parameters meant for `matplotlib.rcParams`. By default, only
365            sets default font to cm serif.
366            See [Matplotlib documentation](https://matplotlib.org/stable/tutorials/introductory/customizing.html)
367            for more information.
368
369        Returns
370        -------
371        An instance of `FisherFigure1D`.
372        """
373        size = len(self.values[0])
374
375        if max_cols is not None and max_cols <= size:
376            full = size % max_cols == 0
377            layout = size // max_cols if full else size // max_cols + 1, max_cols
378        else:
379            layout = 1, size
380            full = True
381
382        if not mpl_options:
383            mpl_options = get_default_rcparams()
384
385        with plt.rc_context(mpl_options):
386            # general figure setup
387            fig = plt.figure(
388                clear=True,
389                figsize=(scale * 2 * layout[1], scale * 2 * layout[0]),
390            )
391            gs = fig.add_gridspec(
392                nrows=layout[0],
393                ncols=layout[1],
394                hspace=0.5,
395                wspace=0.1,
396            )
397            axes = gs.subplots()
398            if size == 1:
399                axes = np.array([axes])
400
401            names = self.values[0].names
402            latex_names = self.values[0].latex_names
403
404            for (index, name), latex_name in zip(enumerate(names), latex_names):
405                ax = axes.flat[index]
406
407                for c, fm in zip(get_default_cycler(), self.values):
408                    plot_curve_1d(fm, name, ax)
409
410                    add_shading_1d(
411                        fm.fiducials[np.where(fm.names == name)],
412                        fm.constraints(name, marginalized=True),
413                        ax,
414                        color=c["color"],
415                        alpha=0.3,
416                        ec=None,
417                    )
418
419                    add_shading_1d(
420                        fm.fiducials[np.where(fm.names == name)],
421                        fm.constraints(name, marginalized=True),
422                        ax,
423                        level=2,
424                        color=c["color"],
425                        alpha=0.1,
426                        ec=None,
427                    )
428
429                ax.set_xlabel(latex_name)
430                ax.relim()
431                ax.autoscale_view()
432
433                # the y axis should start at 0 since we're plotting a PDF
434                ax.set_ylim(0, ax.get_ylim()[-1])
435
436                if index == 0:
437                    ax.set_ylabel(self.ylabel1d)
438
439                ax.set_yticks([])
440
441            if kwargs.get("legend") is True:
442                # remove any axes which are not drawn
443                if not full:
444                    for index in range(
445                        (layout[0] - 1) * layout[1] + 1, layout[0] * layout[1]
446                    ):
447                        axes.flat[index].remove()
448
449            # remove any axes which are not drawn
450            if not full:
451                for index in range(
452                    (layout[0] - 1) * layout[1] + 1, layout[0] * layout[1]
453                ):
454                    axes.flat[index].remove()
455
456        return FisherFigure1D(fig, axes, names)

458    def plot_triangle(
459        self,
460        mpl_options: Optional[dict] = None,
461        plot_1d_curves: bool = True,
462        **kwargs,
463    ):
464        """
465        Plots the 2D contours (and optionally 1D curves) of the Fisher objects.
466
467        Parameters
468        ----------
469        mpl_options : Optional[dict] = None
470            any parameters meant for `matplotlib.rcParams`. By default, only
471            sets default font to cm serif.
472            See [Matplotlib documentation](https://matplotlib.org/stable/tutorials/introductory/customizing.html)
473            for more information.
474        plot_1d_curves : bool = True
475            whether or not the 1D (marginalized) curves should be plotted
476
477        Returns
478        -------
479        An instance of `FisherFigure2D`.
480        """
481        size = len(self.values[0])
482
483        if size < 2:
484            raise ValueError("Unable to make a 2D plot with < 2 parameters")
485
486        if not mpl_options:
487            mpl_options = get_default_rcparams()
488
489        with plt.rc_context(mpl_options):
490            # general figure setup
491            fig = plt.figure(figsize=(2 * size, 2 * size), clear=True)
492            gs = fig.add_gridspec(nrows=size, ncols=size, hspace=0.2, wspace=0.2)
493            # TODO make it work with a shared xcol
494            ax = gs.subplots(sharex="col", sharey=False)
495
496            names = self.values[0].names
497            latex_names = self.values[0].latex_names
498
499            # set automatic limits
500            for i, j in product(range(size), repeat=2):
501                if i == j:
502                    ax[i, i].set_yticks([])
503                    ax[i, i].set_yticklabels([])
504                if i > 0 and 0 < j < size - 1:
505                    ax[i, j].set_yticks([])
506                    ax[i, j].set_yticklabels([])
507
508            for i, j in product(range(len(names)), repeat=2):
509                namey, latex_namey, namex, latex_namex = (
510                    names[i],
511                    latex_names[i],
512                    names[j],
513                    latex_names[j],
514                )
515                # labels for 2D contours (increasing y)
516                if i > 0 and j == 0:
517                    ax[i, j].set_ylabel(latex_namey)
518
519                # labels for 2D contours (increasing x)
520                if i == size - 1:
521                    ax[i, j].set_xlabel(latex_namex)
522
523                # removing any unnecessary axes from the gridspec
524                # TODO should they be removed, or somehow just made invisible?

526                    ax[i, j].remove()

531                        # plot 1-sigma 2D curves

544                        # same thing, but shaded
545                        plot_curve_2d(
546                            fm,
547                            namex,
548                            namey,
549                            ax=ax[i, j],
550                            fill=True,
551                            alpha=0.2,
552                            ec=None,
553                            color=c["color"],
554                            zorder=20,
555                        )

557                        # the 2-sigma

562                            ax=ax[i, j],
563                            scaling_factor=2,
564                            fill=False,
565                            color=c["color"],

582                else:
583                    # plotting the 1D Gaussians

631def get_chisq(
632    sigma: float = 1,
633    df: int = 2,
634):
635    r"""
636    Returns $\Delta \chi^2$.
637    To obtain the scaling coefficient $\alpha$, just take the square root of
638    the output.
639
640    Parameters
641    ----------
642    sigma : float = 1
643        the error on the parameter
644
645    df : int = 2
646        the number of degrees of freedom
647
648    Returns
649    -------
650    float
651    """
652    return chi2.ppf(norm.cdf(sigma) - norm.cdf(-sigma), df=df)

655def add_plot_1d(
656    fiducial: float,
657    sigma: float,
658    ax: Optional[Axes] = None,
659    **kwargs,
660) -> tuple[Figure, Axes]:
661    """
662    Adds a 1D Gaussian with marginalized constraints `sigma` close to fiducial
663    value `fiducial` to axis `ax`.
664    """
665    if ax is None:
666        _, ax = plt.subplots()
667
668    x = np.linspace(
669        fiducial - 4 * sigma,
670        fiducial + 4 * sigma,
671        100,
672    )
673
674    ax.plot(
675        x,
676        [norm.pdf(_, loc=fiducial, scale=sigma) for _ in x],
677        **kwargs,
678    )
679
680    return (ax.get_figure(), ax)

683def add_shading_1d(
684    fiducial: float,
685    sigma: float,
686    ax: Optional[Axes] = None,
687    level: float = 1,
688    **kwargs,
689) -> tuple[Figure, Axes]:
690    """
691    Add shading to a 1D axes.
692    """
693    if ax is None:
694        _, ax = plt.subplots()
695
696    x = np.ndarray.flatten(
697        np.linspace(
698            fiducial - level * sigma,
699            fiducial + level * sigma,
700            100,
701        )
702    )
703
704    ax.fill_between(
705        x,
706        np.ndarray.flatten(
707            np.array([norm.pdf(_, loc=fiducial, scale=sigma) for _ in x])
708        ),
709        **kwargs,
710    )
711
712    return (ax.get_figure(), ax)

715def plot_curve_1d(
716    fisher: FisherMatrix,
717    name: str,
718    ax: Optional[Axes] = None,
719    **kwargs,
720):
721    """
722    Plots a 1D curve (usually marginalized Gaussian) from a Fisher object.
723
724    Parameters
725    --------
726    fisher : FisherMatrix
727        the Fisher matrix which we want to plot
728
729    name : str
730        the name of the parameter which we want to plot
731
732    ax : Optional[matplotlib.axes.Axes] = None
733        the axis on which we want to plot the contour. By default, plots to a
734        new figure.
735
736    Returns
737    -------
738    A 2-tuple `(Figure, Axes)`
739    """
740    if ax is None:
741        _, ax = plt.subplots()
742
743    fid = fisher.fiducials[np.where(fisher.names == name)]
744    sigma = fisher.constraints(name, marginalized=True)
745
746    add_plot_1d(fid, sigma, ax=ax, **kwargs)
747
748    return (ax.get_figure(), ax)

788    fisher: FisherMatrix,
789    name1: str,
790    name2: str,
791    ax: Optional[Axes] = None,
792    scaling_factor: float = 1,
793    **kwargs,
794) -> tuple[Union[None, Figure], Axes]:
795    """
796    Plots a 2D curve (usually ellipse) from two parameters of a Fisher object.
797
798    Parameters
799    ----------
800    fisher : FisherMatrix
801        the Fisher matrix which we want to plot
802
803    name1 : str
804        the name of the first parameter which we want to plot
805

810        the axis on which we want to plot the contour. By default, plots to a
811        new figure.

813    Returns
814    -------

816    """
817    if ax is None:
818        _, ax = plt.subplots()
819

821    fidy = fisher.fiducials[np.where(fisher.names == name2)]

824

830            angle=angle,
831            **kwargs,
832        ),
833    )
834
835    return (ax.get_figure(), ax)

838def get_ellipse(
839    fm: FisherMatrix,
840    name1: str,
841    name2: str,
842) -> tuple[float, float, float]:
843    """
844    Constructs parameters for a Gaussian ellipse from the names.
845    """
846    if name1 == name2:
847        raise ValueError("Names must be different")
848
849    # the inverse
850    inv = FisherMatrix(fm.inverse(), names=fm.names)
851
852    sigmax2 = inv[name1, name1]
853    sigmay2 = inv[name2, name2]
854    sigmaxy = inv[name1, name2]
855
856    # basically the eigenvalues of the submatrix
857    a = np.sqrt(
858        (sigmax2 + sigmay2) / 2 + np.sqrt((sigmax2 - sigmay2) ** 2 / 4 + sigmaxy**2)
859    )
860
861    b = np.sqrt(
862        (sigmax2 + sigmay2) / 2 - np.sqrt((sigmax2 - sigmay2) ** 2 / 4 + sigmaxy**2)
863    )
864
865    angle = np.rad2deg(
866        np.arctan2(
867            2 * sigmaxy,
868            sigmax2 - sigmay2,
869        )
870        / 2
871    )
872
873    return a, b, angle
