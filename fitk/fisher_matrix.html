  55class FisherMatrix:
  56    r"""
  57    Class for handling Fisher objects.
  58
  59    Examples
  60    --------
  61    Specify a Fisher object with default names and fiducials:
  62    >>> fm = FisherMatrix(np.diag([5, 4]), latex_names=['$p_1$', '$p_2$'])
  63
  64    The object has a friendly representation in the interactive session:
  65    >>> fm
  66    FisherMatrix(
  67        array([[5., 0.],
  68           [0., 4.]]),
  69        names=array(['p1', 'p2'], dtype=object),
  70        latex_names=array(['$p_1$', '$p_2$'], dtype=object),
  71        fiducials=array([0., 0.]))
  72
  73    In a Jupyter notebook, it looks similar to the below (you can hover over
  74    the $\LaTeX$ label to show the corresponding name and fiducial of that
  75    parameter):
  76
  77    <table>
  78    <thead>
  79    <tr>
  80    <th></th>
  81    <th title="name = 'p1', fiducial = 0.0">$p_1$</th>
  82    <th title="name = 'p2', fiducial = 0.0">$p_2$</th>
  83    </tr>
  84    </thead>
  85    <tbody>
  86    <tr>
  87    <th title="name = 'p1', fiducial = 0.0">$p_1$</th>
  88    <td>5.000</td>
  89    <td>0.000</td>
  90    </tr>
  91    <tr>
  92    <th title="name = 'p2', fiducial = 0.0">$p_2$</th>
  93    <td>0.000</td>
  94    <td>4.000</td>
  95    </tr>
  96    </tbody>
  97    </table>

  99    List the names:
 100    >>> fm.names
 101    array(['p1', 'p2'], dtype=object)

 104    >>> fm.values
 105    array([[5., 0.],
 106           [0., 4.]])
 107

 113    must of course match the original):

 119        array([[5., 0.],

 123        fiducials=array([0., 0.]))

 215    def __init__(
 216        self,
 217        values,
 218        names: Optional[Collection[str]] = None,
 219        latex_names: Optional[Collection[str]] = None,
 220        fiducials: Optional[Collection[float]] = None,
 221    ):
 222        r"""
 223        Constructor for Fisher object.
 224
 225        Parameters
 226        ----------
 227        values : array_like of float
 228            The values of the Fisher object.
 229
 230        names : array_like of str, optional
 231            The names of the parameters (default: None).
 232            If not specified, defaults to `p1, ..., pn`.
 233
 234        latex_names : array_like of str, optional
 235            The LaTeX names of the parameters (default: None).
 236            If not specified, defaults to `names`.
 237
 238        fiducials : array_like of float, optional
 239            The fiducial values of the parameters (default: None). If not
 240            specified, default to 0 for all parameters.
 241
 242        Raises
 243        ------
 244        ValueError
 245            if the input array has the wrong dimensionality (not 2)
 246
 247        ValueError
 248            if the object is not square-like
 249
 250        MismatchingSizeError
 251            if the sizes of the array of names, values, LaTeX names and
 252            fiducials do not match
 253
 254        Examples
 255        --------
 256        Specify a Fisher object without names:
 257        >>> FisherMatrix(np.diag([1, 2, 3]))
 258        FisherMatrix(
 259            array([[1., 0., 0.],
 260               [0., 2., 0.],
 261               [0., 0., 3.]]),
 262            names=array(['p1', 'p2', 'p3'], dtype=object),
 263            latex_names=array(['p1', 'p2', 'p3'], dtype=object),
 264            fiducials=array([0., 0., 0.]))
 265
 266        Specify a Fisher object with names, LaTeX names, and fiducials:
 267        >>> FisherMatrix(np.diag([1, 2]), names=['alpha', 'beta'],
 268        ... latex_names=[r'$\alpha$', r'$\beta$'], fiducials=[-3, 2])
 269        FisherMatrix(
 270            array([[1., 0.],
 271               [0., 2.]]),
 272            names=array(['alpha', 'beta'], dtype=object),
 273            latex_names=array(['$\\alpha$', '$\\beta$'], dtype=object),
 274            fiducials=array([-3.,  2.]))
 275        """
 276        if np.ndim(values) != 2:
 277            raise ValueError(f"The object {values} is not 2-dimensional")
 278
 279        if not is_square(values):
 280            raise ValueError(f"The object {values} is not square-like")
 281
 282        # try to treat it as an array-like object
 283        self._values = np.array(values, dtype=float)
 284
 285        self._size = np.shape(self._values)[0]
 286        self._ndim = np.ndim(self._values)
 287
 288        # setting the fiducials
 289        if fiducials is None:
 290            self._fiducial = np.zeros(self._size, dtype=float)
 291        else:
 292            self._fiducial = np.array(fiducials, dtype=float)
 293
 294        # setting the names
 295        if names is None:
 296            self._names = make_default_names(self._size)
 297        else:
 298            # check they're unique
 299            if len(set(names)) != len(names):
 300                raise MismatchingSizeError(set(names), names)
 301
 302            self._names = np.array(names, dtype=object)
 303
 304        # setting the pretty names (LaTeX)
 305        if latex_names is None:
 306            self._latex_names = copy.deepcopy(self._names)
 307        else:
 308            self._latex_names = np.array(latex_names, dtype=object)
 309
 310        # check sizes of inputs
 311        if not all(
 312            len(_) == self._size
 313            for _ in (self._names, self._fiducial, self._latex_names)
 314        ):
 315            raise MismatchingSizeError(
 316                self._values[0],
 317                self._names,
 318                self._fiducial,
 319                self._latex_names,
 320            )

 411    def rename(
 412        self,
 413        names: Mapping[str, Union[str, Mapping]],
 414        ignore_errors: bool = False,
 415    ) -> FisherMatrix:
 416        """
 417        Returns a Fisher object with new names. This is primarily useful for
 418        renaming individual parameters.
 419
 420        Parameters
 421        ----------
 422        names : dict-like
 423            a mapping (dictionary-like object) between the old names and the
 424            new ones. The values it maps to can either be a string (the new
 425            name), or a dict with keys `name`, `latex_name`, and `fiducial`
 426            (only `name` is mandatory).
 427
 428        ignore_errors : bool, optional
 429            if set to True, will not raise an error if a parameter doesn't
 430            exist (default: False)
 431
 432        Returns
 433        -------
 434        FisherMatrix
 435            the Fisher object with new names
 436
 437        Examples
 438        --------
 439        >>> m = FisherMatrix(np.diag([1, 2, 3]))
 440        >>> m.rename({'p1' : 'a', 'p2' : dict(name='b', latex_name='$b$', fiducial=2)})
 441        FisherMatrix(
 442            array([[1., 0., 0.],
 443               [0., 2., 0.],
 444               [0., 0., 3.]]),
 445            names=array(['a', 'b', 'p3'], dtype=object),
 446            latex_names=array(['a', '$b$', 'p3'], dtype=object),
 447            fiducials=array([0., 2., 0.]))
 448        """
 449        # check uniqueness and size
 450        if len(set(names)) != len(names):
 451            raise MismatchingSizeError(set(names), names)
 452
 453        if not ignore_errors:
 454            for name in names:
 455                if name not in self.names:
 456                    raise ParameterNotFoundError(name, self.names)
 457
 458        names_new = copy.deepcopy(self.names)
 459        latex_names_new = copy.deepcopy(self.latex_names)
 460        fiducial_new = copy.deepcopy(self.fiducials)
 461
 462        for name, value in names.items():
 463            index = np.where(names_new == name)
 464            # it's a mapping
 465            if isinstance(value, Mapping):
 466                value = _process_fisher_mapping(value)
 467                latex_names_new[index] = value["latex_name"]
 468                fiducial_new[index] = value["fiducial"]
 469                names_new[index] = value["name"]
 470            # otherwise, it's a string
 471            else:
 472                names_new[index] = value
 473                latex_names_new[index] = value
 474
 475        return self.__class__(
 476            self.values,
 477            names=names_new,
 478            latex_names=latex_names_new,
 479            fiducials=fiducial_new,
 480        )

 545    def __getitem__(
 546        self,
 547        keys: Union[tuple[str, ...], slice],
 548    ):
 549        """
 550        Implements access to elements in the Fisher object.
 551        Has support for slicing.
 552        """
 553        # the object can be sliced
 554        if isinstance(keys, slice):
 555            start, stop, step = keys.indices(len(self))
 556            indices = (slice(start, stop, step),) * self.ndim
 557            sl = slice(start, stop, step)
 558            values = self.values[indices]
 559            names = self.names[sl]
 560            latex_names = self.latex_names[sl]
 561            fiducials = self.fiducials[sl]
 562
 563            return self.__class__(
 564                values,
 565                names=names,
 566                latex_names=latex_names,
 567                fiducials=fiducials,
 568            )
 569
 570        try:
 571            _ = iter(keys)
 572        except TypeError as err:
 573            raise TypeError(err) from err
 574
 575        # the keys can be a tuple
 576        if isinstance(keys, tuple):
 577            if len(keys) != self.ndim:
 578                raise ValueError(f"Expected {self.ndim} arguments, got {len(keys)}")
 579
 580            # error checking
 581            for key in keys:
 582                if key not in self.names:
 583                    raise ParameterNotFoundError(key, self.names)
 584
 585            indices = tuple(np.where(self.names == key) for key in keys)
 586
 587        # otherwise, it's some generic object
 588        else:
 589            if keys not in self.names:
 590                raise ParameterNotFoundError(keys, self.names)
 591
 592            indices = (np.where(self.names == keys),)
 593
 594        return self._values[indices][0, 0]

 596    def __setitem__(
 597        self,
 598        keys: tuple[str, ...],
 599        value: float,
 600    ):
 601        """
 602        Implements setting of elements in the Fisher object.
 603        Does not support slicing.
 604
 605        Raises
 606        ------
 607        TypeError
 608            if keys are not a tuple, or if `value` is not a number
 609
 610        MismatchingSizeError
 611            if the size of keys are not equal to the dimensionality of the
 612            Fisher object
 613
 614        ParameterNotFoundError
 615            if any of the keys is not in the parameter names
 616        """
 617        try:
 618            _ = iter(keys)
 619        except TypeError as err:
 620            raise TypeError(err) from err
 621
 622        # the above fails for strings (since they are iterable), so we do an
 623        # explicit check here
 624        if isinstance(keys, str):
 625            raise MismatchingSizeError(keys)
 626
 627        if len(keys) != self.ndim:
 628            raise MismatchingSizeError(keys)
 629
 630        if not isinstance(value, Number):
 631            raise TypeError(f"{value} is not a number")
 632
 633        for key in keys:
 634            if key not in self.names:
 635                raise ParameterNotFoundError(key, self.names)
 636
 637        # automatically raises a value error
 638        indices = tuple(np.where(self.names == key) for key in keys)
 639
 640        temp_data = copy.deepcopy(self._values)
 641
 642        if not all(index == indices[0] for index in indices):
 643            for permutation in list(permutations(indices)):
 644                # update all symmetric parts
 645                temp_data[permutation] = value
 646        else:
 647            temp_data[indices] = value
 648
 649        self._values = copy.deepcopy(temp_data)

 651    def is_valid(self):
 652        """
 653        Checks whether the values make a valid Fisher object.
 654        A (square) matrix is a Fisher matrix if it satisifies the following two
 655        criteria:
 656
 657        * it is symmetric
 658        * it is positive semi-definite
 659
 660        Returns
 661        -------
 662        bool
 663            Whether or nor the Fisher object is valid.
 664
 665        Examples
 666        --------
 667        >>> m = FisherMatrix(np.diag([1, 2, 3]))
 668        >>> m.is_valid()
 669        True
 670        >>> FisherMatrix(np.diag([-1, 3])).is_valid()
 671        False
 672        """
 673        return is_symmetric(self.values) and is_positive_semidefinite(self.values)

 675    def sort(
 676        self,

 679        """
 680        Sorts the Fisher object by name according to some criterion.
 681
 682        Parameters
 683        ----------

 685            all of the other keyword arguments for the Python builtin `sorted`.

 695            values.

 697        Returns
 698        -------
 699        FisherMatrix
 700            the Fisher object with sorted names
 701
 702        Examples
 703        --------
 704        Define a Fisher matrix:
 705        >>> m = FisherMatrix(np.diag([3, 1, 2]), names=list('sdf'),
 706        ... latex_names=['hjkl', 'qwe', 'll'], fiducials=[8, 7, 3])
 707
 708        Sort according to values of the fiducials:
 709        >>> m.sort(key='fiducials')
 710        FisherMatrix(
 711            array([[2., 0., 0.],
 712               [0., 1., 0.],
 713               [0., 0., 3.]]),
 714            names=array(['f', 'd', 's'], dtype=object),
 715            latex_names=array(['ll', 'qwe', 'hjkl'], dtype=object),
 716            fiducials=array([3., 7., 8.]))
 717
 718        Sort according to the LaTeX names (alphabetically):
 719        >>> m.sort(key='latex_names')
 720        FisherMatrix(
 721            array([[3., 0., 0.],
 722               [0., 2., 0.],
 723               [0., 0., 1.]]),
 724            names=array(['s', 'f', 'd'], dtype=object),
 725            latex_names=array(['hjkl', 'll', 'qwe'], dtype=object),
 726            fiducials=array([8., 3., 7.]))
 727        """
 728        allowed_keys = ("fiducials", "latex_names")
 729        # an integer index
 730        if "key" in kwargs and all(hasattr(_, "__index__") for _ in kwargs["key"]):
 731            index = np.array(kwargs["key"], dtype=int)
 732            names = self.names[index]
 733        # either 'fiducials' or 'latex_names'
 734        elif (
 735            "key" in kwargs
 736            and isinstance(kwargs["key"], str)
 737            and kwargs["key"] in allowed_keys
 738        ):
 739            index = np.argsort(getattr(self, kwargs["key"]))
 740            if "reversed" in kwargs and kwargs["reversed"] is True:
 741                index = np.flip(index)
 742            names = self.names[index]
 743        # the names themselves, in any order
 744        elif "key" in kwargs and set(kwargs["key"]) == set(self.names):
 745            index = get_index_of_other_array(self.names, kwargs["key"])
 746            names = self.names[index]
 747        # something that can be passed to `sorted`
 748        else:
 749            if "key" in kwargs and not callable(kwargs["key"]):
 750                raise TypeError(f"`key={kwargs['key']}` is not callable")
 751            names = sorted(self.names, **kwargs)
 752            index = get_index_of_other_array(self.names, names)
 753
 754        latex_names = self.latex_names[index]
 755        fiducials = self.fiducials[index]
 756        values = reindex_array(self.values, index)
 757
 758        return self.__class__(
 759            values,
 760            names=names,
 761            latex_names=latex_names,
 762            fiducials=fiducials,
 763        )

 765    def __eq__(self, other):
 766        """
 767        The equality operator.
 768        Returns `True` if the operands have the following properties:
 769
 770        * are instances of FisherMatrix
 771        * have same names (potentially shuffled)
 772        * have same dimensionality
 773        * have same fiducials (potentially shuffled)
 774        * have same values (potentially shuffled)
 775        """
 776        if set(self.names) != set(other.names):
 777            return False
 778
 779        # index for re-shuffling parameters
 780        index = get_index_of_other_array(self.names, other.names)
 781
 782        return (
 783            isinstance(other, self.__class__)
 784            and self.ndim == other.ndim
 785            and len(self) == len(other)
 786            and set(self.names) == set(other.names)
 787            and np.allclose(
 788                self.fiducials,
 789                other.fiducials[index],
 790            )
 791            and np.allclose(
 792                self.values,
 793                reindex_array(other.values, index),
 794            )
 795        )

 831    def drop(
 832        self,
 833        *names: str,
 834        invert: bool = False,
 835        ignore_errors: bool = False,
 836    ) -> FisherMatrix:
 837        """
 838        Removes parameters from the Fisher object.
 839
 840        Parameters
 841        ----------
 842        names : str
 843            the names of the parameters to drop.
 844            If passing a list or a tuple, make sure to unpack it using the
 845            asterisk (*).
 846
 847        invert : bool, optional
 848            whether to drop all the parameters NOT in names (the complement)
 849            (default: False)
 850
 851        ignore_errors : bool, optional
 852            should non-existing parameters be ignored (default: False)
 853
 854        Returns
 855        -------
 856        FisherMatrix
 857            the Fisher object with removed names
 858
 859        Examples
 860        --------
 861        >>> m = FisherMatrix(np.diag([1, 2, 3]))
 862
 863        Drop `p1` and `p3`:
 864        >>> m.drop('p1', 'p3')
 865        FisherMatrix(
 866            array([[2.]]),
 867            names=array(['p2'], dtype=object),
 868            latex_names=array(['p2'], dtype=object),
 869            fiducials=array([0.]))
 870
 871        Same result, but note the asterisk (`*`):
 872        >>> m.drop(*['p1', 'p3'])
 873        FisherMatrix(
 874            array([[2.]]),
 875            names=array(['p2'], dtype=object),
 876            latex_names=array(['p2'], dtype=object),
 877            fiducials=array([0.]))
 878
 879        Drop everything *except* `p1` and `p3`:
 880        >>> m.drop('p1', 'p3', invert=True)
 881        FisherMatrix(
 882            array([[1., 0.],
 883               [0., 3.]]),
 884            names=array(['p1', 'p3'], dtype=object),
 885            latex_names=array(['p1', 'p3'], dtype=object),
 886            fiducials=array([0., 0.]))
 887        """
 888        if not ignore_errors and not set(names).issubset(set(self.names)):
 889            raise ValueError(
 890                f"The names ({list(names)}) are not a strict subset "
 891                f"of the parameter names in the Fisher object ({self.names}); "
 892                "you can pass `ignore_errors=True` to ignore this error"
 893            )
 894
 895        if ignore_errors:
 896            names = np.array([name for name in names if name in self.names])
 897
 898        if invert is True:
 899            names = set(names) ^ set(self.names)
 900
 901        if set(names) == set(self.names):
 902            raise ValueError("Unable to remove all parameters")
 903
 904        index = [np.array(np.where(self.names == name), dtype=int) for name in names]
 905
 906        values = self.values
 907        for dim in range(self.ndim):
 908            values = np.delete(
 909                values,
 910                index,
 911                axis=dim,
 912            )
 913
 914        fiducials = np.delete(self.fiducials, index)
 915        latex_names = np.delete(self.latex_names, index)
 916        names = np.delete(self.names, index)
 917
 918        return self.__class__(
 919            values,
 920            names=names,
 921            latex_names=latex_names,
 922            fiducials=fiducials,
 923        )

 988    def constraints(
 989        self,
 990        name: Optional[str] = None,
 991        marginalized: bool = True,
 992        sigma: Optional[float] = None,
 993        p: Optional[float] = None,
 994    ):
 995        r"""
 996        Returns the constraints on a parameter as a float, or on all of them
 997        as a numpy array if `name` is not specified.
 998
 999        Parameters
1000        ----------
1001        name : str, optional
1002            the name of the parameter for which we we want the constraints
1003            (default: None)
1004
1005        marginalized : bool, optional
1006            whether we want the marginalized or the unmarginalized
1007            constraints (default: True).
1008
1009        sigma : float, optional
1010            how many sigmas away (default: 1).
1011
1012        p : float, optional
1013            the confidence interval (p-value) (default: None).
1014
1015        Returns
1016        -------
1017        float or array_like of float
1018
1019        Notes
1020        -----
1021        The user should specify either `sigma` or `p`, but not both
1022        simultaneously.
1023        If neither are specified, defaults to `sigma=1`.
1024
1025        The relationship between `p` and `sigma` is defined via:
1026        $$
1027            p(\sigma) = \int\limits_{\mu - \sigma}^{\mu + \sigma}
1028                        f(x, \mu, 1)\, \mathrm{d}x
1029                      = \mathrm{Erf}(\sigma / \sqrt{2})
1030        $$
1031        and therefore the inverse is simply:
1032        $$
1033            \sigma(p) = \sqrt{2}\, \mathrm{Erf}^{-1}(p)
1034        $$
1035        The values of `p` corresponding to 1, 2, 3 `sigma` are roughly
1036        0.683, 0.954, and 0.997, respectively.
1037
1038        Examples
1039        --------
1040        Get (marginalized by default) constraints for all parameters:
1041        >>> m = FisherMatrix([[3, -2], [-2, 5]])
1042        >>> m.constraints()
1043        array([0.67419986, 0.52223297])
1044
1045        Get the unmarginalized constraints instead:
1046        >>> m.constraints(marginalized=False)
1047        array([0.57735027, 0.4472136 ])
1048
1049        Get the unmarginalized constraints for a single parameter:
1050        >>> m.constraints('p1', marginalized=False)
1051        array([0.57735027])
1052
1053        Get the marginalized constraints for a single parameter:
1054        >>> m.constraints('p1', p=0.682689) # p-value roughly equal to 1 sigma
1055        array([0.67419918])
1056        """
1057        if sigma is not None and p is not None:
1058            raise ValueError(
1059                "Cannot specify both `p` and `sigma` simultaneously; "
1060                "please specify at most one of those"
1061            )
1062
1063        if p is not None:
1064            if not 0 < p < 1:
1065                raise ValueError(
1066                    f"The value of `p` {p} is outside of the allowed range (0, 1)"
1067                )
1068            sigma = np.sqrt(2) * erfinv(p)
1069        elif sigma is None:
1070            sigma = 1
1071
1072        if sigma <= 0:
1073            raise ValueError(
1074                f"The value of `sigma` {sigma} is outside of the allowed range (0, infinify)"
1075            )
1076
1077        if marginalized:
1078            inv = self.__class__(
1079                self.inverse(),
1080                names=self.names,
1081                latex_names=self.latex_names,
1082                fiducials=self.fiducials,
1083            )
1084            result = np.sqrt(np.diag(inv.values)) * sigma
1085        else:
1086            result = 1.0 / np.sqrt(np.diag(self.values)) * sigma
1087
1088        if name is not None:
1089            if name in self.names:
1090                return result[np.where(self.names == name)]
1091            raise ParameterNotFoundError(name, self.names)
1092
1093        return result

1341    def reparametrize(
1342        self,
1343        jacobian,
1344        names: Optional[Collection[str]] = None,
1345        latex_names: Optional[Collection[str]] = None,
1346        fiducials: Optional[Collection[float]] = None,
1347    ) -> FisherMatrix:
1348        """
1349        Returns a new Fisher object with parameters `names`, which are
1350        related to the old ones via the transformation `jacobian`.
1351        See the [Wikipedia
1352        article](https://en.wikipedia.org/w/index.php?title=Fisher_information&oldid=1063384000#Reparametrization)
1353        for more information.
1354
1355        Parameters
1356        ----------
1357        transformation : array_like of float
1358            the Jacobian of the transformation
1359
1360        names : array_like of str, optional
1361            list of new names for the Fisher object (default: None). If None,
1362            uses the old names.
1363
1364        latex_names: array_like of str, optional
1365            list of new LaTeX names for the Fisher object (default: None). If
1366            None, and `names` is set, uses those instead, otherwise uses the
1367            old LaTeX names.
1368
1369        fiducials : array_like of float, optional
1370            the new values of the fiducials (default: None). If None, defaults
1371            to old values.
1372
1373        Returns
1374        -------
1375        FisherMatrix
1376            The new Fisher object with the specified names, LaTeX names, and
1377            fiducials.
1378
1379        Examples
1380        --------
1381        >>> fm = FisherMatrix(np.diag([1, 2]))
1382        >>> fm
1383        FisherMatrix(
1384            array([[1., 0.],
1385               [0., 2.]]),
1386            names=array(['p1', 'p2'], dtype=object),
1387            latex_names=array(['p1', 'p2'], dtype=object),
1388            fiducials=array([0., 0.]))
1389
1390        Define a Jacobian:
1391        >>> jac = [[1, 4], [3, 2]]
1392
1393        Reparametrize according to that Jacobian:
1394        >>> fm.reparametrize(jac, names=['a', 'b'])
1395        FisherMatrix(
1396            array([[19., 16.],
1397               [16., 24.]]),
1398            names=array(['a', 'b'], dtype=object),
1399            latex_names=array(['a', 'b'], dtype=object),
1400            fiducials=array([0., 0.]))
1401        """
1402        values = np.transpose(jacobian) @ self.values @ jacobian
1403
1404        if names is not None:
1405            if len(set(names)) != np.shape(jacobian)[-1]:
1406                raise MismatchingSizeError(names)
1407            if latex_names is not None:
1408                if len(set(latex_names)) != np.shape(jacobian)[-1]:
1409                    raise MismatchingSizeError(latex_names)
1410            else:
1411                latex_names = copy.deepcopy(names)
1412        else:
1413            # we don't transform the names
1414            names = copy.deepcopy(self.names)

1627        r"""
1628        Computes the correlation matrix from the Fisher matrix, whose entries
1629        are given by:
1630        $$
1631            \mathsf{P}_{ij} \equiv \frac{\mathsf{C}_{ij}}{\sqrt{\mathsf{C}_{ii} \mathsf{C}_{jj}}}
1632        $$
1633        where $\mathsf{C}_{ij}$ is the $(i, j)$ element of the covariance
1634        matrix (the inverse of the Fisher matrix).
1635
1636        Returns
1637        -------
1638        array_like : float
1639            the correlation matrix as a numpy array
1640        """
1641        inv = self.inverse()
1642        diag = np.diag(inv)
1643
1644        return inv / np.sqrt(np.outer(diag, diag))

 547        self,
 548        keys: Union[tuple[str, ...], slice],
 549    ):
 550        """
 551        Implements access to elements in the Fisher object.
 552        Has support for slicing.
 553        """
 554        # the object can be sliced
 555        if isinstance(keys, slice):
 556            start, stop, step = keys.indices(len(self))
 557            indices = (slice(start, stop, step),) * self.ndim
 558            sl = slice(start, stop, step)
 559            values = self.values[indices]
 560            names = self.names[sl]
 561            latex_names = self.latex_names[sl]
 562            fiducials = self.fiducials[sl]
 563
 564            return self.__class__(
 565                values,
 566                names=names,
 567                latex_names=latex_names,
 568                fiducials=fiducials,
 569            )
 570
 571        try:
 572            _ = iter(keys)
 573        except TypeError as err:
 574            raise TypeError(err) from err
 575
 576        # the keys can be a tuple
 577        if isinstance(keys, tuple):
 578            if len(keys) != self.ndim:
 579                raise ValueError(f"Expected {self.ndim} arguments, got {len(keys)}")
 580
 581            # error checking
 582            for key in keys:
 583                if key not in self.names:
 584                    raise ParameterNotFoundError(key, self.names)
 585
 586            indices = tuple(np.where(self.names == key) for key in keys)
 587
 588        # otherwise, it's some generic object
 589        else:
 590            if keys not in self.names:
 591                raise ParameterNotFoundError(keys, self.names)
 592
 593            indices = (np.where(self.names == keys),)
 594
 595        return self._values[indices][0, 0]

 597    def __setitem__(
 598        self,
 599        keys: tuple[str, ...],
 600        value: float,
 601    ):
 602        """
 603        Implements setting of elements in the Fisher object.
 604        Does not support slicing.
 605
 606        Raises
 607        ------
 608        TypeError
 609            if keys are not a tuple, or if `value` is not a number
 610
 611        MismatchingSizeError
 612            if the size of keys are not equal to the dimensionality of the
 613            Fisher object
 614
 615        ParameterNotFoundError
 616            if any of the keys is not in the parameter names
 617        """
 618        try:
 619            _ = iter(keys)
 620        except TypeError as err:
 621            raise TypeError(err) from err
 622
 623        # the above fails for strings (since they are iterable), so we do an
 624        # explicit check here
 625        if isinstance(keys, str):
 626            raise MismatchingSizeError(keys)
 627
 628        if len(keys) != self.ndim:
 629            raise MismatchingSizeError(keys)
 630
 631        if not isinstance(value, Number):
 632            raise TypeError(f"{value} is not a number")
 633
 634        for key in keys:
 635            if key not in self.names:
 636                raise ParameterNotFoundError(key, self.names)
 637
 638        # automatically raises a value error
 639        indices = tuple(np.where(self.names == key) for key in keys)
 640
 641        temp_data = copy.deepcopy(self._values)
 642
 643        if not all(index == indices[0] for index in indices):
 644            for permutation in list(permutations(indices)):
 645                # update all symmetric parts
 646                temp_data[permutation] = value
 647        else:
 648            temp_data[indices] = value
 649
 650        self._values = copy.deepcopy(temp_data)

 676    def sort(
 677        self,
 678        **kwargs,
 679    ) -> FisherMatrix:
 680        """
 681        Sorts the Fisher object by name according to some criterion.
 682
 683        Parameters
 684        ----------
 685        **kwargs
 686            all of the other keyword arguments for the Python builtin `sorted`.
 687            If none are specified, will sort according to the names of the parameters.
 688            In the special case that the value of the keyword `key` is set to
 689            either `'fiducials'` or `'latex_names'`, it will sort according to those.
 690            In the second special case that the value of the keyword `key` is
 691            set to an array of integers of equal size as the Fisher object, sorts them
 692            according to those instead.
 693            In the third (and final) special case that the value of the keyword
 694            `key` is set to an array strings matching those of parameter names
 695            of the FisherMatrix, the items are sorted according to those
 696            values.
 697
 698        Returns
 699        -------
 700        FisherMatrix
 701            the Fisher object with sorted names
 702
 703        Examples
 704        --------
 705        Define a Fisher matrix:
 706        >>> m = FisherMatrix(np.diag([3, 1, 2]), names=list('sdf'),
 707        ... latex_names=['hjkl', 'qwe', 'll'], fiducials=[8, 7, 3])
 708
 709        Sort according to values of the fiducials:
 710        >>> m.sort(key='fiducials')
 711        FisherMatrix(
 712            array([[2., 0., 0.],
 713               [0., 1., 0.],
 714               [0., 0., 3.]]),
 715            names=array(['f', 'd', 's'], dtype=object),
 716            latex_names=array(['ll', 'qwe', 'hjkl'], dtype=object),
 717            fiducials=array([3., 7., 8.]))
 718
 719        Sort according to the LaTeX names (alphabetically):
 720        >>> m.sort(key='latex_names')
 721        FisherMatrix(
 722            array([[3., 0., 0.],
 723               [0., 2., 0.],
 724               [0., 0., 1.]]),
 725            names=array(['s', 'f', 'd'], dtype=object),
 726            latex_names=array(['hjkl', 'll', 'qwe'], dtype=object),
 727            fiducials=array([8., 3., 7.]))
 728        """
 729        allowed_keys = ("fiducials", "latex_names")
 730        # an integer index
 731        if "key" in kwargs and all(hasattr(_, "__index__") for _ in kwargs["key"]):
 732            index = np.array(kwargs["key"], dtype=int)
 733            names = self.names[index]
 734        # either 'fiducials' or 'latex_names'
 735        elif (
 736            "key" in kwargs
 737            and isinstance(kwargs["key"], str)
 738            and kwargs["key"] in allowed_keys
 739        ):
 740            index = np.argsort(getattr(self, kwargs["key"]))
 741            if "reversed" in kwargs and kwargs["reversed"] is True:
 742                index = np.flip(index)
 743            names = self.names[index]
 744        # the names themselves, in any order
 745        elif "key" in kwargs and set(kwargs["key"]) == set(self.names):
 746            index = get_index_of_other_array(self.names, kwargs["key"])
 747            names = self.names[index]
 748        # something that can be passed to `sorted`
 749        else:
 750            if "key" in kwargs and not callable(kwargs["key"]):
 751                raise TypeError(f"`key={kwargs['key']}` is not callable")
 752            names = sorted(self.names, **kwargs)
 753            index = get_index_of_other_array(self.names, names)
 754
 755        latex_names = self.latex_names[index]
 756        fiducials = self.fiducials[index]
 757        values = reindex_array(self.values, index)
 758
 759        return self.__class__(
 760            values,
 761            names=names,
 762            latex_names=latex_names,
 763            fiducials=fiducials,
 764        )

 766    def __eq__(self, other):
 767        """
 768        The equality operator.
 769        Returns `True` if the operands have the following properties:
 770
 771        * are instances of FisherMatrix
 772        * have same names (potentially shuffled)
 773        * have same dimensionality
 774        * have same fiducials (potentially shuffled)
 775        * have same values (potentially shuffled)
 776        """
 777        if set(self.names) != set(other.names):
 778            return False
 779
 780        # index for re-shuffling parameters
 781        index = get_index_of_other_array(self.names, other.names)
 782
 783        return (
 784            isinstance(other, self.__class__)
 785            and self.ndim == other.ndim
 786            and len(self) == len(other)
 787            and set(self.names) == set(other.names)
 788            and np.allclose(
 789                self.fiducials,
 790                other.fiducials[index],
 791            )
 792            and np.allclose(
 793                self.values,
 794                reindex_array(other.values, index),
 795            )
 796        )

 832    def drop(
 833        self,
 834        *names: str,
 835        invert: bool = False,
 836        ignore_errors: bool = False,
 837    ) -> FisherMatrix:
 838        """
 839        Removes parameters from the Fisher object.
 840
 841        Parameters
 842        ----------
 843        names : str
 844            the names of the parameters to drop.
 845            If passing a list or a tuple, make sure to unpack it using the
 846            asterisk (*).
 847
 848        invert : bool, optional
 849            whether to drop all the parameters NOT in names (the complement)
 850            (default: False)
 851
 852        ignore_errors : bool, optional
 853            should non-existing parameters be ignored (default: False)
 854
 855        Returns
 856        -------
 857        FisherMatrix
 858            the Fisher object with removed names
 859
 860        Examples
 861        --------
 862        >>> m = FisherMatrix(np.diag([1, 2, 3]))
 863
 864        Drop `p1` and `p3`:
 865        >>> m.drop('p1', 'p3')
 866        FisherMatrix(
 867            array([[2.]]),
 868            names=array(['p2'], dtype=object),
 869            latex_names=array(['p2'], dtype=object),
 870            fiducials=array([0.]))
 871
 872        Same result, but note the asterisk (`*`):
 873        >>> m.drop(*['p1', 'p3'])
 874        FisherMatrix(
 875            array([[2.]]),
 876            names=array(['p2'], dtype=object),
 877            latex_names=array(['p2'], dtype=object),
 878            fiducials=array([0.]))
 879
 880        Drop everything *except* `p1` and `p3`:
 881        >>> m.drop('p1', 'p3', invert=True)
 882        FisherMatrix(
 883            array([[1., 0.],
 884               [0., 3.]]),
 885            names=array(['p1', 'p3'], dtype=object),
 886            latex_names=array(['p1', 'p3'], dtype=object),
 887            fiducials=array([0., 0.]))
 888        """
 889        if not ignore_errors and not set(names).issubset(set(self.names)):
 890            raise ValueError(
 891                f"The names ({list(names)}) are not a strict subset "
 892                f"of the parameter names in the Fisher object ({self.names}); "
 893                "you can pass `ignore_errors=True` to ignore this error"
 894            )
 895
 896        if ignore_errors:
 897            names = np.array([name for name in names if name in self.names])
 898
 899        if invert is True:
 900            names = set(names) ^ set(self.names)
 901
 902        if set(names) == set(self.names):
 903            raise ValueError("Unable to remove all parameters")
 904
 905        index = [np.array(np.where(self.names == name), dtype=int) for name in names]
 906
 907        values = self.values
 908        for dim in range(self.ndim):
 909            values = np.delete(
 910                values,
 911                index,
 912                axis=dim,
 913            )
 914
 915        fiducials = np.delete(self.fiducials, index)
 916        latex_names = np.delete(self.latex_names, index)
 917        names = np.delete(self.names, index)
 918
 919        return self.__class__(
 920            values,
 921            names=names,
 922            latex_names=latex_names,
 923            fiducials=fiducials,
 924        )

 989    def constraints(
 990        self,
 991        name: Optional[str] = None,
 992        marginalized: bool = True,
 993        sigma: Optional[float] = None,
 994        p: Optional[float] = None,
 995    ):
 996        r"""
 997        Returns the constraints on a parameter as a float, or on all of them
 998        as a numpy array if `name` is not specified.
 999
1000        Parameters
1001        ----------
1002        name : str, optional
1003            the name of the parameter for which we we want the constraints
1004            (default: None)
1005
1006        marginalized : bool, optional
1007            whether we want the marginalized or the unmarginalized
1008            constraints (default: True).
1009
1010        sigma : float, optional
1011            how many sigmas away (default: 1).
1012
1013        p : float, optional
1014            the confidence interval (p-value) (default: None).
1015
1016        Returns
1017        -------
1018        float or array_like of float
1019
1020        Notes
1021        -----
1022        The user should specify either `sigma` or `p`, but not both
1023        simultaneously.
1024        If neither are specified, defaults to `sigma=1`.
1025
1026        The relationship between `p` and `sigma` is defined via:
1027        $$
1028            p(\sigma) = \int\limits_{\mu - \sigma}^{\mu + \sigma}
1029                        f(x, \mu, 1)\, \mathrm{d}x
1030                      = \mathrm{Erf}(\sigma / \sqrt{2})
1031        $$
1032        and therefore the inverse is simply:
1033        $$
1034            \sigma(p) = \sqrt{2}\, \mathrm{Erf}^{-1}(p)
1035        $$
1036        The values of `p` corresponding to 1, 2, 3 `sigma` are roughly
1037        0.683, 0.954, and 0.997, respectively.
1038
1039        Examples
1040        --------
1041        Get (marginalized by default) constraints for all parameters:
1042        >>> m = FisherMatrix([[3, -2], [-2, 5]])
1043        >>> m.constraints()
1044        array([0.67419986, 0.52223297])
1045
1046        Get the unmarginalized constraints instead:
1047        >>> m.constraints(marginalized=False)
1048        array([0.57735027, 0.4472136 ])
1049
1050        Get the unmarginalized constraints for a single parameter:
1051        >>> m.constraints('p1', marginalized=False)
1052        array([0.57735027])
1053
1054        Get the marginalized constraints for a single parameter:
1055        >>> m.constraints('p1', p=0.682689) # p-value roughly equal to 1 sigma
1056        array([0.67419918])
1057        """
1058        if sigma is not None and p is not None:
1059            raise ValueError(
1060                "Cannot specify both `p` and `sigma` simultaneously; "
1061                "please specify at most one of those"
1062            )
1063
1064        if p is not None:
1065            if not 0 < p < 1:
1066                raise ValueError(
1067                    f"The value of `p` {p} is outside of the allowed range (0, 1)"
1068                )
1069            sigma = np.sqrt(2) * erfinv(p)
1070        elif sigma is None:
1071            sigma = 1
1072
1073        if sigma <= 0:
1074            raise ValueError(
1075                f"The value of `sigma` {sigma} is outside of the allowed range (0, infinify)"
1076            )
1077
1078        if marginalized:
1079            inv = self.__class__(
1080                self.inverse(),
1081                names=self.names,
1082                latex_names=self.latex_names,
1083                fiducials=self.fiducials,
1084            )
1085            result = np.sqrt(np.diag(inv.values)) * sigma
1086        else:
1087            result = 1.0 / np.sqrt(np.diag(self.values)) * sigma
1088
1089        if name is not None:
1090            if name in self.names:
1091                return result[np.where(self.names == name)]
1092            raise ParameterNotFoundError(name, self.names)
1093
1094        return result

1342    def reparametrize(
1343        self,
1344        jacobian,
1345        names: Optional[Collection[str]] = None,
1346        latex_names: Optional[Collection[str]] = None,
1347        fiducials: Optional[Collection[float]] = None,
1348    ) -> FisherMatrix:
1349        """
1350        Returns a new Fisher object with parameters `names`, which are
1351        related to the old ones via the transformation `jacobian`.
1352        See the [Wikipedia
1353        article](https://en.wikipedia.org/w/index.php?title=Fisher_information&oldid=1063384000#Reparametrization)
1354        for more information.
1355
1356        Parameters
1357        ----------
1358        transformation : array_like of float
1359            the Jacobian of the transformation
1360
1361        names : array_like of str, optional
1362            list of new names for the Fisher object (default: None). If None,
1363            uses the old names.
1364
1365        latex_names: array_like of str, optional
1366            list of new LaTeX names for the Fisher object (default: None). If
1367            None, and `names` is set, uses those instead, otherwise uses the
1368            old LaTeX names.
1369
1370        fiducials : array_like of float, optional
1371            the new values of the fiducials (default: None). If None, defaults
1372            to old values.
1373
1374        Returns
1375        -------
1376        FisherMatrix
1377            The new Fisher object with the specified names, LaTeX names, and
1378            fiducials.
1379
1380        Examples
1381        --------
1382        >>> fm = FisherMatrix(np.diag([1, 2]))
1383        >>> fm
1384        FisherMatrix(
1385            array([[1., 0.],
1386               [0., 2.]]),
1387            names=array(['p1', 'p2'], dtype=object),
1388            latex_names=array(['p1', 'p2'], dtype=object),
1389            fiducials=array([0., 0.]))
1390
1391        Define a Jacobian:
1392        >>> jac = [[1, 4], [3, 2]]
1393
1394        Reparametrize according to that Jacobian:
1395        >>> fm.reparametrize(jac, names=['a', 'b'])
1396        FisherMatrix(
1397            array([[19., 16.],
1398               [16., 24.]]),
1399            names=array(['a', 'b'], dtype=object),
1400            latex_names=array(['a', 'b'], dtype=object),
1401            fiducials=array([0., 0.]))
1402        """
1403        values = np.transpose(jacobian) @ self.values @ jacobian
1404
1405        if names is not None:
1406            if len(set(names)) != np.shape(jacobian)[-1]:
1407                raise MismatchingSizeError(names)
1408            if latex_names is not None:
1409                if len(set(latex_names)) != np.shape(jacobian)[-1]:
1410                    raise MismatchingSizeError(latex_names)
1411            else:
1412                latex_names = copy.deepcopy(names)
1413        else:
1414            # we don't transform the names
1415            names = copy.deepcopy(self.names)
1416            latex_names = copy.deepcopy(self.latex_names)
1417
1418        if fiducials is not None:
1419            if len(fiducials) != np.shape(jacobian)[-1]:
1420                raise MismatchingSizeError(fiducials)
1421        else:
1422            fiducials = copy.deepcopy(self.fiducials)
1423
1424        return self.__class__(
1425            values,
1426            names=names,
1427            latex_names=latex_names,
1428            fiducials=fiducials,
1429        )
