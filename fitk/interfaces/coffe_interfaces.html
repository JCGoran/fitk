 30def _parse_and_set_args(
 31    **kwargs,
 32):
 33    """
 34    Parses kwargs for COFFE, and returns an instance of it.
 35    """
 36    # these can't be set using `setattr(<instance>, <name>, <value>)`,
 37    # so we need to process them separately
 38    special_kwargs = (
 39        "galaxy_bias1",
 40        "galaxy_bias2",
 41        "magnification_bias1",
 42        "magnification_bias2",
 43    )
 44
 45    bias = {}
 46    for key, value in kwargs.items():
 47        if key in special_kwargs:
 48            bias[key] = value
 49
 50    result = coffe.Coffe(
 51        **{key: value for key, value in kwargs.items() if key not in special_kwargs}
 52    )
 53
 54    if bias:
 55        for key, value in bias.items():
 56            # basically equivalent to:
 57            # <instance>.set_<thing>_bias{1, 2}(x_array, y_array)
 58            getattr(result, f"set_{key}")(value[0], value[1])
 59
 60    return result

158    def covariance(
159        self,
160        *args: tuple[str, float],
161        **kwargs,
162    ):
163        r"""
164        Computes the covariance of the multipoles of the 2PCF with some
165        cosmology.
166
167        Returns
168        -------
169        array_like : float
170            the covariance matrix as a numpy array
171
172        Notes
173        -----
174        The covariance does not take into account cross-correlations between
175        the different redshifts, i.e. for $n$ redshift bins it has the form:
176        $$
177            \begin{pmatrix}
178            \mathsf{C}(\bar{z}_1) & 0 & \ldots & 0\\\
179            0 & \mathsf{C}(\bar{z}_2) & \ldots & 0\\\
180            \vdots & \vdots & \ddots & \vdots\\\
181            0 & 0 & \ldots & \mathsf{C}(\bar{z}_n)
182            \end{pmatrix}
183        $$
184
185        For more details on the exact theoretical modelling used, see
186        [arXiv:1806.11090](https://arxiv.org/abs/1806.11090), section 2.
187        """
188        cosmo = _parse_and_set_args(**self.config)
189        for arg, value in args:
190            setattr(cosmo, arg, value)
191
192        covariance = cosmo.compute_covariance_bulk()
193
194        result = block_diag(
195            *[
196                np.reshape(
197                    [
198                        _.value
199                        for _ in covariance[
200                            i : i + len(cosmo.sep) ** 2 * len(cosmo.l) ** 2
201                        ]
202                    ],
203                    (len(cosmo.sep) * len(cosmo.l), len(cosmo.sep) * len(cosmo.l)),
204                )
205                for i in range(
206                    0,
207                    len(covariance),
208                    len(cosmo.sep) ** 2 * len(cosmo.l) ** 2,
209                )
210            ]
211        )
212
213        return result

225class CoffeMultipolesTildeDerivative(CoffeMultipolesDerivative):
226    r"""
227    Class for computing the derivatives of the 2PCF w.r.t. $\tilde{f}$ and
228    $\tilde{b}$ parametrization, which are defined as:
229
230    $$
231        \tilde{f}(z) = f(z) \sigma_8(z),
232        \quad
233        \tilde{b}(z) = b(z) \sigma_8(z)
234    $$
235
236    where $f(z)$ is the growth rate of matter, $b(z)$ is the linear galaxy
237    bias, and $\sigma_8(z)$ is the size of the redshift-dependent matter
238    perturbation at 8 $\mathrm{Mpc}/h$.
239
240    The valid parameters are:
241    * $\tilde{f}$ - `f[N]`, the parameter $\sigma_8(z) D_1(z)$ in a redshift
242    bin, where `[N]` is a number from 1 to the number of redshift bins
243    * $\tilde{b}$ - `b[N]`, the parameter $b(z) D_1(z)$ in a redshift bin,
244    where `[N]` is a number from 1 to the number of redshift bins
245
246    Notes
247    -----
248    Just like for `CoffeMultipolesDerivative`, one can set the configuration in
249    the constructor by specifying the `config` argument.
250
251    Only density and RSD effects are taken into account when computing the
252    signal.
253
254    The covariance is assumed to be parameter-independent.
255
256    If the biases are set for two populations, the second population is ignored.
257    """
258
259    def validate_parameter(self, arg) -> bool:
260        if re.search(r"^[bf][1-9][0-9]*$", arg.name) and int(arg.name[1:]) in range(
261            1, len(self.config["z_mean"]) + 1
262        ):
263            return True
264        return False

273    def signal(
274        self,
275        *args: tuple[str, float],
276        **kwargs,
277    ):
278        cosmo = _parse_and_set_args(**self.config)
279
280        def monopole(b: float, f: float):
281            return (
282                (b**2 + 2 * b * f / 3 + f**2 / 5)
283                * np.array([cosmo.integral(r=_, l=0, n=0) for _ in cosmo.sep])
284                / cosmo.sigma8**2
285            )
286
287        def quadrupole(b: float, f: float):
288            return (
289                -(4 * b * f / 3 + 4 * f**2 / 7)
290                * np.array([cosmo.integral(r=_, l=2, n=0) for _ in cosmo.sep])
291                / cosmo.sigma8**2
292            )
293
294        def hexadecapole(b: float, f: float):
295            return (
296                (8 * f**2 / 35)
297                * np.array([cosmo.integral(r=_, l=4, n=0) for _ in cosmo.sep])
298                / cosmo.sigma8**2
299            )
300
301        multipoles = {0: monopole, 2: quadrupole, 4: hexadecapole}
302
303        b_array = np.array(
304            [
305                cosmo.galaxy_bias1(_) * cosmo.growth_factor(_) * cosmo.sigma8
306                for _ in cosmo.z_mean
307            ]
308        )
309        f_array = np.array(
310            [
311                cosmo.growth_rate(_) * cosmo.growth_factor(_) * cosmo.sigma8
312                for _ in cosmo.z_mean
313            ]
314        )
315
316        for arg in args:
317            name, value = arg
318            if name[0] == "b":
319                b_array[int(name[1:]) - 1] = value
320            elif name[0] == "f":
321                f_array[int(name[1:]) - 1] = value
322
323        return np.array(
324            [
325                np.array(
326                    [
327                        [
328                            item(b, f)
329                            for key, item in multipoles.items()
330                            if key in cosmo.l
331                        ]
332                        for b, f in zip(b_array, f_array)
333                    ]
334                )
335            ]
336        ).flatten()
