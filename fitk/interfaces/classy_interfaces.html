 99    def _parse_outputs(self):
100        raw_output = self.config.get("output", "")
101
102        if isinstance(raw_output, (tuple, list, np.ndarray)):
103            outputs = raw_output
104        else:
105            outputs = {item.strip() for item in raw_output.split(",")}
106
107        return {
108            "temperature": "tCl" in outputs,
109            "polarization": "pCl" in outputs,

142class ClassyCMBDerivative(ClassyBaseDerivative):
143    """
144    Interface for CMB quantities.
145
146    Interface for computing derivatives using the CMB signal and covariance
147    (temperature, polarization, or both).
148    """
149
150    def __init__(
151        self,
152        *args,
153        config: Optional[dict] = None,
154        **kwargs,
155    ):
156        """
157        Create an instance.
158
159        Parameters
160        ----------
161        config : dict, optional
162            the CLASS configuration to use. All parameters are accepted.
163            If not specified, defaults to `{'output' : 'tCl'}`.
164            If the key `output` is missing, it is inserted with a default value
165            `tCl`.
166        """
167        super().__init__(*args, config=config, **kwargs)
168        self._config = config if config is not None else {"output": "tCl"}
169        if not self._config.get("output"):
170            self._config["output"] = "tCl"
171
172    def signal(
173        self,
174        *args: tuple[str, float],
175        **kwargs,
176    ):
177        r"""
178        Compute the CMB $C_\ell$s.
179
180        Parameters
181        ----------
182        *args
183            the name(s) and value(s) of the parameter(s) for which we want to
184            compute the derivative
185
186        Returns
187        -------
188        result : array_like of float
189            the signal as a numpy array
190
191        Notes
192        -----
193        The signal is composed of the following contributions, in this order:
194        - temperature $C_\ell$s if `output` contains `tCl`
195        - E-mode $C_\ell$s if `output` contains `pCl`
196        - cross-correlations between T and E-modes if `output` contains `tCl`
197          and `pCl`
198        - B-mode $C_\ell$s if `output` contains `pCl` *and they are non-zero*
199
200        Note that $\ell = \\{0, 1\\}$ are not part of the output, i.e. we impose
201        $\ell_\mathrm{min} = 2$.
202
203        The output is ordered as follows:
204
205        $$
206            \mathbf{S} = \\{C_\ell^{TT}, C_\ell^{EE}, C_\ell^{TE}, C_\ell^{BB} \\}
207        $$
208        """
209        cosmo = self._run_classy(*args)
210
211        outputs = self._parse_outputs()
212
213        result = []
214        if outputs["temperature"]:
215            result.extend(cosmo.raw_cl()["tt"][2:])
216        if outputs["polarization"]:
217            result.extend(cosmo.raw_cl()["ee"][2:])
218        if outputs["temperature"] and outputs["polarization"]:
219            result.extend(cosmo.raw_cl()["te"][2:])
220        if outputs["polarization"] and np.any(cosmo.raw_cl()["bb"]):
221            result.extend(cosmo.raw_cl()["bb"][2:])
222
223        if result:
224            return np.array(result)
225
226        return NotImplemented
227
228    def _prefactor_covariance(self, size: int):
229        l_max = int(self.config.get("l_max_scalars", 2500))
230        matrix = np.diag([2 / (2 * ell + 1) for ell in range(2, l_max + 1)])
231        return np.tile(matrix, (size, size))
232
233    def covariance(
234        self,
235        *args: tuple[str, float],
236        **kwargs,
237    ):
238        r"""
239        Compute the covariance of CMB $C_\ell$s.
240
241        Parameters
242        ----------
243        *args
244            the name(s) and fiducial value(s) of the parameter(s) for which we want to
245            compute the covariance
246
247        **kwargs
248            keyword arguments for the covariance. Supported values are:
249            - `fsky`: the sky fraction of the survey (default: 1)
250
251        Returns
252        -------
253        result : array_like of float
254            the signal as a numpy array
255
256        Notes
257        -----
258        The covariance is the following block-matrix (with the notation $X = C_\ell$):
259
260        $$
261            \frac{2}{2 \ell + 1}
262            \begin{pmatrix}
263            (X^{TT})^2 & (X^{TE})^2 & X^{TT} X^{TE} & 0 \\\\
264            (X^{TE})^2 & (X^{EE})^2 & X^{EE} X^{TE} & 0 \\\\
265            X^{TT} X^{TE} & X^{EE} X^{TE} & [(X^{TE})^2 + X^{TT} X^{EE}] / 2 & 0 \\\\
266            0 & 0 & 0 & (X^{BB})^2
267            \end{pmatrix}
268        $$
269
270        See the notes of the `signal` method about the order of the outputs in
271        the matrix.
272
273        The covariance has been taken from <a
274        href="https://arxiv.org/abs/0911.3105" target="_blank" rel="noreferrer
275        noopener">arXiv:0911.3105</a>, eq. (27).
276        """
277        cosmo = self._run_classy(*args)
278
279        outputs = self._parse_outputs()
280
281        c_ell = cosmo.raw_cl()
282        c_tt = c_ell.get("tt", [])[2:]
283        c_ee = c_ell.get("ee", [])[2:]
284        c_te = c_ell.get("te", [])[2:]
285        c_bb = c_ell.get("bb", [])[2:]
286
287        # CMB C_ells
288        if outputs["temperature"] and outputs["polarization"]:
289            result = np.block(
290                [
291                    [np.diag(c_tt**2), np.diag(c_te**2), np.diag(c_tt * c_te)],
292                    [np.diag(c_te**2), np.diag(c_ee**2), np.diag(c_ee * c_te)],
293                    [
294                        np.diag(c_tt * c_te),
295                        np.diag(c_ee * c_te),
296                        np.diag(c_te**2 + c_tt * c_ee) / 2,
297                    ],
298                ]
299            )
300
301            if np.any(c_bb):
302                result = self._prefactor_covariance(4) * block_diag(
303                    result,
304                    np.diag(c_bb**2),
305                )
306            else:
307                result = self._prefactor_covariance(3) * result
308
309        elif outputs["temperature"]:
310            result = self._prefactor_covariance(1) * np.diag(c_tt**2)
311
312        elif outputs["polarization"]:
313            if np.any(c_bb):
314                result = self._prefactor_covariance(2) * block_diag(
315                    np.diag(c_ee**2), np.diag(c_bb**2)
316                )
317            else:
318                result = self._prefactor_covariance(1) * np.diag(c_ee**2)
319
320        else:
321            return NotImplemented
322
323        final_kwargs = self._parse_covariance_kwargs(**kwargs)
324
325        return result / final_kwargs["fsky"]

420    def derivative(
421        self,
422        method: str,
423        *args: D,
424        rounding_threshold: float = 0.0,
425        **kwargs,
426    ):
427        r"""
428        Evaluate the derivative of `method` with respect to arguments `args`.
429
430        Parameters
431        ----------
432        method : {'signal', 'covariance'}
433            the object (method) for which we want to compute the derivative
434
435        *args
436            the parameters (see description of `D`) for which we want to
437            compute the derivatives
438
439        rounding_threshold, optional
440            the threshold for rounding the derivative, in case one knows from
441            other considerations that it should be zero (default: 0)
442
443        **kwargs
444            any keyword arguments passed to the method
445
446        Returns
447        -------
448        array_like of float
449            the resulting derivative as a numpy array
450
451        Raises
452        ------
453        ValueError
454            if `method` is not one of {'signal', 'covariance'}
455        """
456        # check that `method` is valid
457        valid_methods = ["signal", "covariance"]
458        if method not in valid_methods:
459            raise ValueError(
460                f"The method `{method}` is not one of: {valid_methods}",
461            )
462        _validate_derivatives(*args)
463
464        if rounding_threshold < 0:
465            raise ValueError(
466                "The value of the argument `rounding_threshold` must be positive or zero"
467            )
468
469        parsed_args = _parse_derivatives(*args)
470
471        weights_arr = [
472            find_diff_weights(arg.stencil, order=arg.order) for arg in parsed_args
473        ]
474        points_arr = [arg.stencil for arg in parsed_args]
475
476        denominator = np.prod([arg.abs_step**arg.order for arg in parsed_args])
477
478        stencils = np.array(list(product(*points_arr)))
479        weights = np.array([np.prod(_) for _ in product(*weights_arr)])
480
481        # zero-out elements which are close to zero (they are non-zero due to
482        # floating-point math)
483        weights = np.array([_ if np.abs(_) > 1e-10 else 0 for _ in weights])
484
485        # remove them from the stencil
486        stencils = stencils[np.nonzero(weights)]
487        weights = weights[np.nonzero(weights)]
488
489        return _zero_out(
490            np.array(
491                np.sum(
492                    [
493                        getattr(self, method)(
494                            *[
495                                (
496                                    arg.parameter.name,
497                                    arg.parameter.fiducial + arg.abs_step * p,
498                                )
499                                for arg, p in zip(parsed_args, point)
500                            ],
501                            **kwargs,
502                        )
503                        * weight
504                        / denominator
505                        for point, weight in zip(stencils, weights)
506                    ],
507                    axis=0,
508                )
509            ),
510            rounding_threshold,
511        )

529            the parameters (see description of `D`) for which we want to
530            compute the derivatives
531
532        parameter_dependence : {'signal', 'covariance', 'both'}
533            where the parameter dependence is located, in the signal, the
534            covariance, or both (default: 'signal')
535
536        use_pinv : bool, optional
537            use the Moore-Penrose pseudoinverse to obtain the inverse of the
538            covariance matrix (default: False). Can be useful if the covariance
539            matrix is weakly signular. Use with caution!
540
541        external_covariance : array_like, optional
542            the external covariance matrix to use when computing the result
543            (default: None)
544
545        rounding_threshold, optional
546            the threshold for rounding the derivative, in case one knows from
547            other considerations that it should be zero (default: 0)
548
549        kwargs_signal : dict, optional
550            any keyword arguments that should be passed to `signal` (default:
551            None)
552
553        kwargs_covariance : dict, optional
554            any keyword arguments that should be passed to `covariance`
555            (default: None)
556
557        Returns
558        -------
559        FisherMatrix
560            the Fisher object with corresponding names and fiducials
561
562        Warns
563        -----
564        UserWarning
565            if `parameter_dependence` is set to 'covariance' or 'both', and
566            `external_covariance` is not `None`
567
568        Raises
569        ------
570        NotImplementedError
571            if either `signal` or `covariance` have not been implemented, and
572            the user set `parameter_dependence` to be in one of those
573
574        ValidationError
575            if the parameter validation failed
576
577        ValueError
578            if the argument `external_covariance` is not a square matrix
579
580        LinAlgError
581            if the covariance matrix is singular
582
583        Notes
584        -----
585        The `order` parameter is ignored if passed to `D`.
586
587        The element $\mathsf{F}_{ij}$ for parameters $(\theta_i, \theta_j)$ is
588        defined as:
589        $$
590            \frac{\partial \mathbf{S}^T}{\partial \theta_i}
591            \mathsf{C}^{-1}
592            \frac{\partial \mathbf{S}}{\partial \theta_j}
593            +
594            \frac{1}{2}
595            \mathrm{Tr}\left(
596            \frac{\partial \mathsf{C}}{\partial \theta_i}
597            \mathsf{C}^{-1}
598            \frac{\partial \mathsf{C}}{\partial \theta_j}
599            \mathsf{C}^{-1}
600            \right)
601        $$
602        where $\mathbf{S}$ is the signal vector, $\mathsf{C}$ is the covariance
603        matrix, and $\mathrm{Tr}(X)$ denotes the trace of the quantity $X$.
604        """
605        # first we attempt to compute the covariance; if that fails, it means
606        # it hasn't been implemented, so we fail fast and early
607        names = np.array([_.parameter.name for _ in args])
608        fiducials = np.array([_.parameter.fiducial for _ in args])
609        latex_names = np.array([_.parameter.latex_name for _ in args])
610
611        if kwargs_signal is None:
612            kwargs_signal = {}
613
614        if kwargs_covariance is None:
615            kwargs_covariance = {}
616
617        for arg in args:
618            if not self.validate_parameter(arg.parameter):
619                raise ValidationError(arg.parameter)
620
621        if external_covariance is not None:
622            if not (
623                np.ndim(external_covariance) == 2
624                and len(set(np.shape(external_covariance))) == 1
625            ):
626                raise ValueError(
627                    "The argument `external_covariance` must be a square matrix"
628                )
629
630            if (
631                parameter_dependence in ["covariance", "both"]
632                and external_covariance is not None
633            ):
634                warnings.warn(
635                    "`external_covariance` has no effect if `parameter_dependence` "
636                    f"is set to '{parameter_dependence}'"
637                )
638                covariance_matrix = self.covariance(
639                    *zip(names, fiducials), **kwargs_covariance
640                )
641
642            else:
643                covariance_matrix = external_covariance
644        else:
645            covariance_matrix = self.covariance(
646                *zip(names, fiducials), **kwargs_covariance
647            )
648
649        if use_pinv:
650            inverse_covariance_matrix = np.linalg.pinv(
651                covariance_matrix,
652                hermitian=True,
653            )
654        else:
655            inverse_covariance_matrix = np.linalg.inv(covariance_matrix)
656
657        covariance_shape = np.shape(inverse_covariance_matrix)
658
659        signal_derivative = {}
660
661        # TODO parallelize
662        for arg in args:
663            if parameter_dependence in ["signal", "both"]:
664                signal_derivative[arg.parameter.name] = self.derivative(
665                    "signal",
666                    D(
667                        parameter=arg.parameter,
668                        abs_step=arg.abs_step,
669                        kind=arg.kind,
670                        accuracy=arg.accuracy,
671                        stencil=arg.stencil,
672                    ),
673                    rounding_threshold=rounding_threshold,
674                    **kwargs_signal,
675                )
676
677        covariance_derivative = {}
678
679        # TODO parallelize
680        for arg in args:
681            if parameter_dependence in ["covariance", "both"]:
682                covariance_derivative[arg.parameter.name] = self.derivative(
683                    "covariance",
684                    D(
685                        parameter=arg.parameter,
686                        abs_step=arg.abs_step,
687                        kind=arg.kind,
688                        accuracy=arg.accuracy,
689                        stencil=arg.stencil,
690                    ),
691                    rounding_threshold=rounding_threshold,
692                    **kwargs_covariance,
693                )
694            else:
695                covariance_derivative[arg.parameter.name] = np.zeros(covariance_shape)
696
697        fisher_matrix = np.zeros([len(args)] * 2)
698
699        for (i, arg1), (j, arg2) in product(enumerate(args), repeat=2):
700            fisher_matrix[i, j] = matrix_element_from_input(
701                inverse_covariance_matrix,
702                signal_derivative1=signal_derivative[arg1.parameter.name],
703                signal_derivative2=signal_derivative[arg2.parameter.name],
704                covariance_derivative1=covariance_derivative[arg1.parameter.name],
705                covariance_derivative2=covariance_derivative[arg2.parameter.name],
706            )
707
708        return FisherMatrix(
709            fisher_matrix,
710            names=names,
711            fiducials=fiducials,
712            latex_names=latex_names,
713        )
