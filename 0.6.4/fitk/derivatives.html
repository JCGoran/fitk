 24def _zero_out(array, threshold: float):
 25    """
 26    Returns the input array with elements smaller than `threshold` zeroed out.
 27    """
 28    # Create a copy of the input array
 29    new_array = np.copy(array)
 30
 31    # Find the elements in the array that are below the threshold
 32    below_threshold = np.abs(new_array) < threshold
 33
 34    # Replace those elements with 0
 35    new_array[below_threshold] = 0
 36
 37    return new_array

 40def _validate_derivatives(
 41    *args: D,
 42):
 43    """
 44    Checks whether we are able to compute the requested derivatives
 45
 46    Raises
 47    ------
 48    ValueError
 49        if we are unable to compute the requested derivatives
 50    """
 51    upper_limit = 10
 52
 53    if np.sum([arg.order for arg in args]) > upper_limit:
 54        raise ValueError(
 55            f"Unable to compute derivatives with combined order > {upper_limit}"
 56        )

 59def _parse_derivatives(*args: D):
 60    """
 61    Parses the derivatives
 62    """
 63    parsed_args: list[D] = []
 64    for arg in args:
 65        # in case the argument repeats, we just increase the order of the
 66        # derivative
 67        if arg in parsed_args:
 68            parsed_args[parsed_args.index(arg)].order += arg.order
 69        else:
 70            parsed_args.append(arg)
 71
 72    return tuple(parsed_args)

 75def matrix_element_from_input(
 76    inverse_covariance: Collection[Collection[float]],
 77    signal_derivative1: Optional[Collection[float]] = None,
 78    signal_derivative2: Optional[Collection[float]] = None,
 79    covariance_derivative1: Optional[Collection[Collection[float]]] = None,
 80    covariance_derivative2: Optional[Collection[Collection[float]]] = None,
 81):
 82    """
 83    Computes the element of a Fisher matrix given the raw derivative arrays
 84
 85    Parameters
 86    ----------
 87    inverse_covariance : array_like of float
 88        the inverse of the covariance matrix of the system
 89
 90    signal_derivative1 : array_like of float, optional
 91        the derivative of the signal w.r.t. the first parameter
 92
 93    signal_derivative2 : array_like of float, optional
 94        the derivative of the signal w.r.t. the second parameter
 95
 96    covariance_derivative1 : array_like of float, optional
 97        the derivative of the covariance w.r.t. the first parameter
 98
 99    covariance_derivative2 : array_like of float, optional
100        the derivative of the covariance w.r.t. the second parameter
101
102    Returns
103    -------
104    float
105        the element of the Fisher matrix as a float
106
107    Raises
108    ------
109    ValueError
110        if the inputs have mismatching sizes
111
112    ValueError
113        if all of the parameters (`signal_derivative1`,
114        `signal_derivative2`, `covariance_derivative1`,
115        `covariance_derivative2`) are `None`
116
117    ValueError
118        if only one of the two derivatives (`signal_derivative1` or
119        `signal_derivative2`, `covariance_derivative1` or
120        `covariance_derivative2`) is not `None`
121
122    Notes
123    -----
124    The signal and the inverse covariance must be shape-compatible.
125
126    This convenience method is useful if one has already computed the
127    derivatives, and wishes to get the Fisher matrix element without using
128    `FisherDerivative` directly.
129    """
130    signal_derivative1 = np.array(signal_derivative1)
131    signal_derivative2 = np.array(signal_derivative2)
132    covariance_derivative1 = np.array(covariance_derivative1)
133    covariance_derivative2 = np.array(covariance_derivative2)
134    inverse_covariance = np.array(inverse_covariance)
135
136    if (
137        not signal_derivative1.any()
138        and not signal_derivative2.any()
139        and not covariance_derivative1.any()
140        and not covariance_derivative2.any()
141    ):
142        raise ValueError(
143            "All four inputs (`signal_derivative1`, `signal_derivative2`, "
144            "`covariance_derivative1`, `covariance_derivative2`) are unspecified, "
145            "nothing to compute"
146        )
147
148    if (signal_derivative1.any() and not signal_derivative2.any()) or (
149        not signal_derivative1.any() and signal_derivative2.any()
150    ):
151        raise ValueError(
152            "Either both `signal_derivative1` and `signal_derivative2` should be None, "
153            "or neither of them should be None"
154        )
155
156    if (covariance_derivative1.any() and not covariance_derivative2.any()) or (
157        not covariance_derivative1.any() and covariance_derivative2.any()
158    ):
159        raise ValueError(
160            "Either both `covariance_derivative1` and `covariance_derivative2` should be None, "
161            "or neither of them should be None"
162        )
163
164    # part which contains signal dependance on the parameters
165    signal_dependence = 0
166
167    if signal_derivative1.any():
168        signal_dependence = signal_derivative1 @ inverse_covariance @ signal_derivative2
169
170    # part which contains covariance dependance on the parameters
171    covariance_dependence = 0
172
173    if covariance_derivative1.any():
174        covariance_dependence = (
175            np.trace(
176                inverse_covariance
177                @ covariance_derivative1
178                @ inverse_covariance
179                @ covariance_derivative2
180            )
181            / 2
182        )
183
184    return covariance_dependence + signal_dependence

188class D:
189    """
190    Class for describing information about a derivative of a parameter using
191    finite differences
192
193    Parameters
194    ----------
195    name
196        the name of the parameter w.r.t. which we take the derivative
197
198    fiducial
199        the point where we want to compute the derivative
200
201    abs_step
202        the absolute step size for computing the derivative
203
204    order, optional
205        the order of the derivative (default: 1)
206
207    accuracy, optional
208        the accuracy requested for the derivative (default: 4)
209
210    kind : {'center', 'forward', 'backward'}
211        the kind of difference to use (default: 'center')
212
213    stencil, optional
214        the custom stencil used for computing the derivative (default: None).
215        If specified, the arguments `accuracy` and `kind` are ignored.
216
217    latex_name, optional
218        the display name of the parameter. If not specified, equals to name
219        (default: None)
220
221    Raises
222    ------
223    ValueError
224        is raised in one of the following situations:
225        * if the value of `abs_step` is not positive
226        * if the value of `stencil` is not strictly monotonically
227        increasing
228        * if the value of `accuracy` is not at least 1
229        * if the value of `kind` is not one of: 'center', 'forward', 'backward'
230
231    TypeError
232        if the value of `stencil` is not an iterable
233    """
234
235    name: str
236    fiducial: float
237    abs_step: float
238    order: int = 1
239    accuracy: int = 4
240    kind: str = "center"
241    stencil: Optional[Collection[float]] = None
242    latex_name: Optional[str] = None
243
244    def __eq__(self, other):
245        return (
246            isinstance(other, self.__class__)
247            and self.name == other.name
248            and np.allclose(self.fiducial, other.fiducial)
249            and np.allclose(self.abs_step, other.abs_step)
250            and np.allclose(self.order, other.order)
251            and self.kind == other.kind
252            and np.allclose(self.stencil, other.stencil)
253        )
254
255    def __post_init__(self):
256        if self.abs_step <= 0:
257            raise ValueError("The value of `abs_step` must be positive")
258
259        # if we specify a stencil, we don't bother checking `kind` and
260        # `accuracy`
261        if self.stencil is not None:
262            if not is_iterable(self.stencil):
263                raise TypeError(
264                    f"The value of the argument `stencil` ({self.stencil}) is not an iterable"
265                )
266
267            if not np.allclose(sorted(self.stencil), self.stencil):
268                raise ValueError(
269                    f"The value of the argument `stencil` ({self.stencil}) "
270                    "should be an increasing list of numbers"
271                )
272
273            return
274
275        allowed_kinds = ["center", "forward", "backward"]
276
277        if self.kind not in allowed_kinds:
278            raise ValueError(
279                f"Only the following derivative kinds are allowed: {allowed_kinds}"
280            )
281
282        if self.accuracy < 1:
283            raise ValueError("The accuracy must be at least first-order")
284
285        if self.kind == "center":
286            npoints = (2 * np.floor((self.order + 1) / 2) - 2 + self.accuracy) // 2
287            self.stencil = np.arange(-npoints, npoints + 1, 1)
288        elif self.kind == "forward":
289            npoints = self.accuracy + self.order
290            self.stencil = np.arange(0, npoints + 1)
291        else:
292            npoints = self.accuracy + self.order
293            self.stencil = np.arange(-npoints, 1)
294
295        if not self.latex_name:
296            self.latex_name = self.name

398    def derivative(
399        self,
400        method: str,
401        *args: D,
402        rounding_threshold: float = 0.0,
403        **kwargs,
404    ):
405        r"""
406        Evaluates the derivative of `method` with respect to arguments `args`.
407
408        Parameters
409        ----------
410        method : {'signal', 'covariance'}
411            the object (method) for which we want to compute the derivative
412
413        *args
414            the parameters (see description of `D`) for which we want to
415            compute the derivatives
416
417        rounding_threshold, optional
418            the threshold for rounding the derivative, in case one knows from
419            other considerations that it should be zero (default: 0)
420
421        **kwargs
422            any keyword arguments passed to the method
423
424        Returns
425        -------
426        array_like of float
427            the resulting derivative as a numpy array
428
429        Raises
430        ------
431        ValueError
432            if `method` is not one of {'signal', 'covariance'}
433        """
434        # check that `method` is valid
435        valid_methods = ["signal", "covariance"]
436        if method not in valid_methods:
437            raise ValueError(
438                f"The method `{method}` is not one of: {valid_methods}",
439            )
440        _validate_derivatives(*args)
441
442        if rounding_threshold < 0:
443            raise ValueError(
444                "The value of the argument `rounding_threshold` must be positive or zero"
445            )
446
447        parsed_args = _parse_derivatives(*args)
448
449        weights_arr = [
450            find_diff_weights(arg.stencil, order=arg.order) for arg in parsed_args
451        ]
452        points_arr = [arg.stencil for arg in parsed_args]
453
454        denominator = np.prod([arg.abs_step**arg.order for arg in parsed_args])
455
456        stencils = np.array(list(product(*points_arr)))
457        weights = np.array([np.prod(_) for _ in product(*weights_arr)])
458
459        # zero-out elements which are close to zero (they are non-zero due to
460        # floating-point math)
461        weights = np.array([_ if np.abs(_) > 1e-10 else 0 for _ in weights])
462
463        # remove them from the stencil
464        stencils = stencils[np.nonzero(weights)]
465        weights = weights[np.nonzero(weights)]
466
467        return _zero_out(
468            np.array(
469                np.sum(
470                    [
471                        getattr(self, method)(
472                            *[
473                                (arg.name, arg.fiducial + arg.abs_step * p)
474                                for arg, p in zip(parsed_args, point)
475                            ],
476                            **kwargs,
477                        )
478                        * weight
479                        / denominator
480                        for point, weight in zip(stencils, weights)
481                    ],
482                    axis=0,
483                )
484            ),
485            rounding_threshold,
486        )

488    def fisher_matrix(
489        self,
490        *args: D,
491        parameter_dependence: str = "signal",
492        external_covariance: Optional[Sequence[Sequence[float]]] = None,
493        rounding_threshold: float = 0.0,
494        kwargs_signal: Optional[dict] = None,
495        kwargs_covariance: Optional[dict] = None,
496    ) -> FisherMatrix:
497        r"""
498        Computes the Fisher matrix, $\mathsf{F}$, using finite differences.
499        The element $\mathsf{F}_{ij}$ for parameters $(\theta_i, \theta_j)$ is
500        defined as:
501        $$
502            \frac{\partial \mathbf{S}^T}{\partial \theta_i}
503            \mathsf{C}^{-1}
504            \frac{\partial \mathbf{S}}{\partial \theta_j}
505            +
506            \frac{1}{2}
507            \mathrm{Tr}\left(
508            \frac{\partial \mathsf{C}}{\partial \theta_i}
509            \mathsf{C}^{-1}
510            \frac{\partial \mathsf{C}}{\partial \theta_j}
511            \mathsf{C}^{-1}
512            \right)
513        $$
514        where $\mathbf{S}$ is the signal vector, $\mathsf{C}$ is the covariance
515        matrix, and $\mathrm{Tr}(X)$ denotes the trace of the quantity $X$.
516
517        Parameters
518        ----------
519        *args
520            the parameters (see description of `D`) for which we want to
521            compute the derivatives
522
523        parameter_dependence : {'signal', 'covariance', 'both'}
524            where the parameter dependence is located, in the signal, the
525            covariance, or both (default: 'signal')
526
527        external_covariance : array_like, optional
528            the external covariance matrix to use when computing the result
529            (default: None)
530
531        rounding_threshold, optional
532            the threshold for rounding the derivative, in case one knows from
533            other considerations that it should be zero (default: 0)
534
535        kwargs_signal : dict, optional
536            any keyword arguments that should be passed to `signal` (default:
537            None)
538
539        kwargs_covariance : dict, optional
540            any keyword arguments that should be passed to `covariance`
541            (default: None)
542
543        Returns
544        -------
545        FisherMatrix
546            the Fisher object with corresponding names and fiducials
547
548        Warns
549        -----
550        UserWarning
551            if `parameter_dependence` is set to 'covariance' or 'both', and
552            `external_covariance` is not `None`
553
554        Raises
555        ------
556        NotImplementedError
557            if either `signal` or `covariance` have not been implemented, and
558            the user set `parameter_dependence` to be in one of those
559
560        ValidationError
561            if the parameter validation failed
562
563        ValueError
564            if the argument `external_covariance` is not a square matrix
565
566        Notes
567        -----
568        The `order` parameter is ignored if passed to `D`.
569        """
570        # first we attempt to compute the covariance; if that fails, it means
571        # it hasn't been implemented, so we fail fast and early
572        names = np.array([_.name for _ in args])
573        fiducials = np.array([_.fiducial for _ in args])
574        latex_names = np.array([_.latex_name for _ in args])
575
576        if kwargs_signal is None:
577            kwargs_signal = {}
578
579        if kwargs_covariance is None:
580            kwargs_covariance = {}
581
582        for arg in args:
583            if not self.validate_parameter(arg):
584                raise ValidationError(arg)
585
586        if external_covariance is not None:
587            if not (
588                np.ndim(external_covariance) == 2
589                and len(set(np.shape(external_covariance))) == 1
590            ):
591                raise ValueError(
592                    "The argument `external_covariance` must be a square matrix"
593                )
594
595            if (
596                parameter_dependence in ["covariance", "both"]
597                and external_covariance is not None
598            ):
599                warnings.warn(
600                    "`external_covariance` has no effect if `parameter_dependence` "
601                    f"is set to '{parameter_dependence}'"
602                )
603                covariance_matrix = self.covariance(
604                    *zip(names, fiducials), **kwargs_covariance
605                )
606
607            else:
608                covariance_matrix = external_covariance
609        else:
610            covariance_matrix = self.covariance(
611                *zip(names, fiducials), **kwargs_covariance
612            )
613
614        inverse_covariance_matrix = np.linalg.inv(covariance_matrix)
615
616        covariance_shape = np.shape(inverse_covariance_matrix)
617
618        signal_derivative = {}
619
620        # TODO parallelize
621        for arg in args:
622            if parameter_dependence in ["signal", "both"]:
623                signal_derivative[arg.name] = self.derivative(
624                    "signal",
625                    D(
626                        name=arg.name,
627                        fiducial=arg.fiducial,
628                        abs_step=arg.abs_step,
629                        kind=arg.kind,
630                        accuracy=arg.accuracy,
631                        stencil=arg.stencil,
632                    ),
633                    rounding_threshold=rounding_threshold,
634                    **kwargs_signal,
635                )
636
637        covariance_derivative = {}
638
639        # TODO parallelize
640        for arg in args:
641            if parameter_dependence in ["covariance", "both"]:
642                covariance_derivative[arg.name] = self.derivative(
643                    "covariance",
644                    D(
645                        name=arg.name,
646                        fiducial=arg.fiducial,
647                        abs_step=arg.abs_step,
648                        kind=arg.kind,
649                        accuracy=arg.accuracy,
650                        stencil=arg.stencil,
651                    ),
652                    rounding_threshold=rounding_threshold,
653                    **kwargs_covariance,
654                )
655            else:
656                covariance_derivative[arg.name] = np.zeros(covariance_shape)
657
658        fisher_matrix = np.zeros([len(args)] * 2)
659
660        for (i, arg1), (j, arg2) in product(enumerate(args), repeat=2):
661            fisher_matrix[i, j] = matrix_element_from_input(
662                inverse_covariance_matrix,
663                signal_derivative1=signal_derivative[arg1.name],
664                signal_derivative2=signal_derivative[arg2.name],
665                covariance_derivative1=covariance_derivative[arg1.name],
666                covariance_derivative2=covariance_derivative[arg2.name],
667            )
668
669        return FisherMatrix(
670            fisher_matrix,
671            names=names,
672            fiducials=fiducials,
673            latex_names=latex_names,
674        )

499        Computes the Fisher matrix, $\mathsf{F}$, using finite differences.
500        The element $\mathsf{F}_{ij}$ for parameters $(\theta_i, \theta_j)$ is
501        defined as:
502        $$
503            \frac{\partial \mathbf{S}^T}{\partial \theta_i}
504            \mathsf{C}^{-1}
505            \frac{\partial \mathbf{S}}{\partial \theta_j}
506            +
507            \frac{1}{2}
508            \mathrm{Tr}\left(
509            \frac{\partial \mathsf{C}}{\partial \theta_i}
510            \mathsf{C}^{-1}
511            \frac{\partial \mathsf{C}}{\partial \theta_j}
512            \mathsf{C}^{-1}
513            \right)
514        $$
515        where $\mathbf{S}$ is the signal vector, $\mathsf{C}$ is the covariance
516        matrix, and $\mathrm{Tr}(X)$ denotes the trace of the quantity $X$.
517
518        Parameters
519        ----------
520        *args
521            the parameters (see description of `D`) for which we want to
522            compute the derivatives
523

526            covariance, or both (default: 'signal')
527

529            the external covariance matrix to use when computing the result
530            (default: None)
531
532        rounding_threshold, optional

534            other considerations that it should be zero (default: 0)
535

538            None)

572        # it hasn't been implemented, so we fail fast and early
573        names = np.array([_.name for _ in args])
574        fiducials = np.array([_.fiducial for _ in args])
575        latex_names = np.array([_.latex_name for _ in args])
576
577        if kwargs_signal is None:
578            kwargs_signal = {}
579

581            kwargs_covariance = {}
582

584            if not self.validate_parameter(arg):
585                raise ValidationError(arg)
586

589                np.ndim(external_covariance) == 2
590                and len(set(np.shape(external_covariance))) == 1
591            ):
592                raise ValueError(
593                    "The argument `external_covariance` must be a square matrix"
594                )
595
596            if (
597                parameter_dependence in ["covariance", "both"]
598                and external_covariance is not None

602                    f"is set to '{parameter_dependence}'"

605                    *zip(names, fiducials), **kwargs_covariance
606                )
607

609                covariance_matrix = external_covariance
610        else:
611            covariance_matrix = self.covariance(
612                *zip(names, fiducials), **kwargs_covariance
613            )
614
615        inverse_covariance_matrix = np.linalg.inv(covariance_matrix)
616

636                )

642            if parameter_dependence in ["covariance", "both"]:
643                covariance_derivative[arg.name] = self.derivative(
644                    "covariance",

Computes the Fisher matrix, $\mathsf{F}$, using finite differences. The element $\mathsf{F}_{ij}$ for parameters $(\theta_i, \theta_j)$ is defined as: $$ \frac{\partial \mathbf{S}^T}{\partial \theta_i} \mathsf{C}^{-1} \frac{\partial \mathbf{S}}{\partial \theta_j} + \frac{1}{2} \mathrm{Tr}\left( \frac{\partial \mathsf{C}}{\partial \theta_i} \mathsf{C}^{-1} \frac{\partial \mathsf{C}}{\partial \theta_j} \mathsf{C}^{-1} \right) $$ where $\mathbf{S}$ is the signal vector, $\mathsf{C}$ is the covariance matrix, and $\mathrm{Tr}(X)$ denotes the trace of the quantity $X$.
