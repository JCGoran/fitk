 616        self,
 617        name: str,
 618        method: str,
 619        *args,
 620        **kwargs,
 621    ):
 622        """
 623        Implements drawing of other objects on the axis.
 624
 625        Parameters
 626        ----------
 627        name : str
 628            the name (label) of the parameter where we want to draw
 629
 630        method : str
 631            the name of the method which we want to plot (such as `plot`,
 632            `scatter`, etc.)
 633
 634        *args
 635            any positional arguments passed to the method (usually some data)
 636
 637        **kwargs
 638            any keyword arguments passed to the method (usually styling options)
 639
 640        Returns
 641        -------
 642        Artist
 643            the artist that was drawn on the figure
 644
 645        Raises
 646        ------
 647        EmptyFigureError
 648            if `plot` hasn't been called yet
 649        """
 650        if not hasattr(self[name], method):
 651            raise AttributeError(
 652                f"The method `{method}` is not a valid plotting method"
 653            )
 654
 655        # get the right color
 656        if "c" not in kwargs and "color" not in kwargs:
 657            kwargs["color"] = next(self.current_color)
 658            self.current_color = iter(self.current_color)
 659
 660        # hopefully the return type should be some artist, or a collection of
 661        # artists
 662        handles = getattr(self[name], method)(*args, **kwargs)
 663        if is_iterable(handles):
 664            for handle in handles:
 665                if isinstance(handle, Artist) and kwargs.get("label"):
 666                    self.labels.append(kwargs.get("label"))
 667                    self.handles.append(handle)
 668        else:
 669            if isinstance(handles, Artist) and kwargs.get("label"):
 670                self.labels.append(kwargs.get("label"))
 671                self.handles.append(handles)
 672
 673        self[name].autoscale()
 674        self[name].relim()
 675        self[name].autoscale_view()
 676
 677        return handles

 679    def plot(
 680        self,
 681        fisher: FisherMatrix,
 682        *args,
 683        **kwargs,
 684    ):
 685        """
 686        Plots the 1D curves of the Fisher objects.
 687
 688        Parameters
 689        ----------
 690        fisher
 691            the Fisher object which we want to plot
 692
 693        **kwargs
 694            any keyword arguments used for plotting (same as `plot_curve_1d`)
 695
 696        Returns
 697        -------
 698        None
 699        """
 700        size = len(fisher)
 701
 702        added = False
 703
 704        # in order to preserve colors for each object, we only use the
 705        # (default) cycler when the color is not explicitly set
 706        if "c" not in kwargs and "color" not in kwargs:
 707            kwargs["color"] = next(self.current_color)
 708            self.current_color = iter(self.current_color)
 709
 710        if self.max_cols is not None and self.max_cols <= size:
 711            full = size % self.max_cols == 0
 712            layout = (
 713                size // self.max_cols if full else size // self.max_cols + 1,
 714                self.max_cols,
 715            )
 716        else:
 717            layout = 1, size
 718            full = True
 719
 720        with plt.rc_context(self.options):
 721            # general figure setup
 722            if self.figure:
 723                fig = self.figure
 724                axes = self.axes
 725            else:
 726                fig = plt.figure(
 727                    clear=True,
 728                    figsize=(2 * layout[1], 2 * layout[0]),
 729                )
 730                gs = fig.add_gridspec(
 731                    nrows=layout[0],
 732                    ncols=layout[1],
 733                    hspace=self.hspace,
 734                    wspace=self.wspace,
 735                )
 736                axes = gs.subplots()
 737                if size == 1:
 738                    axes = np.array([axes])
 739
 740            for (index, name), latex_name in zip(
 741                enumerate(fisher.names),
 742                fisher.latex_names,
 743            ):
 744                ax = axes.flat[index]
 745
 746                _, __, handle = plot_curve_1d(fisher, name, ax, **kwargs)
 747
 748                if not added and kwargs.get("label"):
 749                    self.handles.append(handle)
 750                    self.labels.append(kwargs["label"])
 751                    added = True
 752
 753                _add_shading_1d(
 754                    fisher.fiducials[np.where(fisher.names == name)],
 755                    fisher.constraints(name, marginalized=True),
 756                    ax,
 757                    alpha=0.3,
 758                    ec=None,
 759                    **kwargs,
 760                )
 761
 762                _add_shading_1d(
 763                    fisher.fiducials[np.where(fisher.names == name)],
 764                    fisher.constraints(name, marginalized=True),
 765                    ax,
 766                    level=2,
 767                    alpha=0.1,
 768                    ec=None,
 769                    **kwargs,
 770                )
 771
 772                ax.autoscale()
 773                ax.set_xlabel(latex_name)
 774                ax.relim()
 775                ax.autoscale_view()
 776
 777                # the y axis should start at 0 since we're plotting a PDF
 778                ax.set_ylim(0, ax.get_ylim()[-1])
 779
 780                ax.set_yticks([])
 781
 782            # remove any axes which are not drawn
 783            if not full:
 784                for index in range(
 785                    size,
 786                    layout[0] * layout[1],
 787                ):
 788                    # it's simpler to turn off the axis, than completely remove
 789                    # it
 790                    axes.flat[index].axis("off")
 791
 792        self._figure = fig
 793        self._axes = axes
 794        self._names = fisher.names

1197    def plot(
1198        self,
1199        fisher: FisherMatrix,
1200        *args,
1201        **kwargs,
1202    ):
1203        r"""
1204        Plots the 2D contours (and optionally 1D curves) of the Fisher objects.
1205
1206        Parameters
1207        ----------
1208        fisher : FisherMatrix
1209            the Fisher object which we want to plot
1210
1211        **kwargs
1212            any keyword arguments used for plotting (same as `plot_curve_2d`)
1213
1214        Returns
1215        -------
1216        None
1217        """
1218        size = len(fisher)
1219
1220        if size < 2:
1221            raise ValueError("Unable to make a 2D plot with < 2 parameters")
1222
1223        if self.names is not None:
1224            # make sure they have the same names; if not, raise an error
1225            if set(self.names) != set(fisher.names):
1226                raise ValueError(
1227                    f"The Fisher object names ({fisher.names}) "
1228                    f"do not match those on the figure ({self.names})"
1229                )
1230
1231            # otherwise, we reshuffle them
1232            fisher = fisher.sort(key=self.names)
1233
1234        # in order to preserve colors for each object, we only use the
1235        # (default) cycler when the color is not explicitly set
1236        if "c" not in kwargs and "color" not in kwargs:
1237            kwargs["color"] = next(self.current_color)
1238            self.current_color = iter(self.current_color)
1239
1240        with plt.rc_context(self.options):
1241            # general figure setup
1242            if self.figure:
1243                fig = self.figure
1244                ax = self.axes
1245            else:
1246                fig = plt.figure(
1247                    figsize=(2 * size, 2 * size),
1248                    clear=True,
1249                )
1250                ax = fig.add_gridspec(
1251                    nrows=size,
1252                    ncols=size,
1253                    hspace=self.hspace,
1254                    wspace=self.wspace,
1255                ).subplots(
1256                    sharex="col",
1257                    sharey=False,
1258                )
1259
1260            # set automatic limits
1261            for i, j in product(range(size), repeat=2):
1262                if i == j:
1263                    ax[i, i].set_yticks([])
1264                    ax[i, i].set_yticklabels([])
1265                    if i < size - 1:
1266                        ax[i, i].get_xaxis().set_visible(False)
1267                if i > 0 and 0 < j < size - 1:
1268                    ax[i, j].set_yticks([])
1269                    ax[i, j].set_yticklabels([])
1270                if i > j and i != size - 1:
1271                    ax[i, j].get_xaxis().set_visible(False)
1272
1273            # flag for whether the current artist has already been added to the
1274            # legend handle
1275            added = False
1276            # loop over the parameters
1277            for i, j in product(range(len(fisher.names)), repeat=2):
1278                namey, latex_namey, namex, latex_namex = (
1279                    fisher.names[i],
1280                    fisher.latex_names[i],
1281                    fisher.names[j],
1282                    fisher.latex_names[j],
1283                )
1284                # labels for 2D contours (increasing y)
1285                if i > 0 and j == 0:
1286                    ax[i, j].set_ylabel(latex_namey)
1287
1288                # labels for 2D contours (increasing x)
1289                if i == size - 1:
1290                    ax[i, j].set_xlabel(latex_namex)
1291
1292                # removing any unnecessary axes from the gridspec
1293                if i < j and self.figure is None:
1294                    # calling `remove()` is better than `axis("off")` because
1295                    # then we obtain the proper bounding box
1296                    ax[i, j].remove()
1297
1298                # plotting the 2D contours
1299                elif i > j:
1300                    # plot 1-sigma 2D curves
1301                    # NOTE this is the "68% of the probability of a
1302                    # single parameter lying within the bounds projected
1303                    # onto a parameter axis"
1304                    _, __, handle = plot_curve_2d(
1305                        fisher,
1306                        namex,
1307                        namey,
1308                        ax=ax[i, j],
1309                        scaling_factor=1
1310                        if not self.show_joint_dist
1311                        else np.sqrt(_get_chisq(1)),
1312                        fill=False,
1313                        zorder=20,
1314                        **kwargs,
1315                    )
1316
1317                    if not added and kwargs.get("label"):
1318                        self.handles.append(handle)
1319                        self.labels.append(kwargs["label"])
1320                        added = True
1321
1322                    # same thing, but shaded
1323                    plot_curve_2d(
1324                        fisher,
1325                        namex,
1326                        namey,
1327                        ax=ax[i, j],
1328                        scaling_factor=1
1329                        if not self.show_joint_dist
1330                        else np.sqrt(_get_chisq(1)),
1331                        fill=True,
1332                        alpha=0.2,
1333                        ec=None,
1334                        zorder=20,
1335                        **kwargs,
1336                    )
1337
1338                    # the 2-sigma
1339                    plot_curve_2d(
1340                        fisher,
1341                        namex,
1342                        namey,
1343                        ax=ax[i, j],
1344                        scaling_factor=2
1345                        if not self.show_joint_dist
1346                        else np.sqrt(_get_chisq(2)),
1347                        fill=False,
1348                        zorder=20,
1349                        **kwargs,
1350                    )
1351                    # same thing, but shaded
1352                    plot_curve_2d(
1353                        fisher,
1354                        namex,
1355                        namey,
1356                        ax=ax[i, j],
1357                        scaling_factor=2
1358                        if not self.show_joint_dist
1359                        else np.sqrt(_get_chisq(2)),
1360                        fill=True,
1361                        alpha=0.1,
1362                        ec=None,
1363                        zorder=20,
1364                        **kwargs,
1365                    )
1366
1367                if i == j:
1368                    # plotting the 1D Gaussians
1369                    if self.show_1d_curves is True:
1370                        plot_curve_1d(
1371                            fisher,
1372                            namex,
1373                            ax=ax[i, i],
1374                            **kwargs,
1375                        )
1376
1377                    else:
1378                        if self.figure is None:
1379                            ax[i, i].remove()
1380
1381            # set automatic limits
1382            for i, j in product(range(size), repeat=2):
1383                try:
1384                    ax[i, j].relim()
1385                    ax[i, j].autoscale_view()
1386                    if i == j:
1387                        ax[i, i].autoscale()
1388                        ax[i, i].set_ylim(0, ax[i, i].get_ylim()[-1])
1389                        ax[i, i].set_yticks([])
1390                        ax[i, i].set_yticklabels([])
1391                except AttributeError:
1392                    pass
1393
1394        self._figure = fig
1395        self._axes = ax
1396        self._names = fisher.names

1399def _get_chisq(
1400    sigma: float = 1,
1401    df: int = 2,
1402):
1403    r"""
1404    Returns $\Delta \chi^2$.
1405    To obtain the scaling coefficient $\alpha$, just take the square root of
1406    the output.
1407
1408    Parameters
1409    ----------
1410    sigma : float = 1
1411        the error on the parameter
1412
1413    df : int = 2
1414        the number of degrees of freedom
1415
1416    Returns
1417    -------
1418    float
1419    """
1420    return chi2.ppf(norm.cdf(sigma) - norm.cdf(-sigma), df=df)
