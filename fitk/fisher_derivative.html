 24def _validate_derivatives(
 25    *args: D,
 26):
 27    """
 28    Checks whether we are able to compute the requested derivatives
 29
 30    Raises
 31    ------
 32    * `ValueError` if we are unable to compute the requested derivatives
 33    """
 34    upper_limit = 10
 35

 37        raise ValueError(

 42@dataclass
 43class D:
 44    """
 45    Class for describing information about a derivative of a parameter using
 46    finite differences
 47
 48    Parameters
 49    ----------
 50    name
 51        the name of the parameter w.r.t. which we take the derivative
 52
 53    value
 54        the point where we want to compute the derivative
 55
 56    abs_step
 57        the absolute step size for computing the derivative
 58
 59    order
 60        the order of the derivative (default: 1)
 61
 62    accuracy
 63        the accuracy requested for the derivative (default: 4)
 64
 65    kind
 66        the kind of difference to use (default: 'center'). Available options
 67        are 'center', 'forward', 'backward'.
 68
 69    stencil
 70        the custom stencil used for computing the derivative (default: None).
 71        If specified, the arguments `accuracy` and `kind` are ignored.
 72
 73    Raises
 74    ------
 75    * `ValueError` if the value of `abs_step` is not positive
 76    * `TypeError` if the value of `stencil` is not an iterable
 77    * `ValueError` if the value of `stencil` is not strictly monotonically
 78    increasing
 79    * `ValueError` if the value of `accuracy` is not at least 1
 80    * `ValueError` if the value of `kind` is not one of: 'center', 'forward',
 81    'backward'
 82    """
 83
 84    name: str
 85    value: float
 86    abs_step: float
 87    order: int = 1
 88    accuracy: int = 4
 89    kind: str = "center"
 90    stencil: Optional[Collection[float]] = None
 91
 92    def __post_init__(self):
 93        if self.abs_step <= 0:
 94            raise ValueError("The value of `abs_step` must be positive")
 95
 96        # if we specify a stencil, we don't bother checking `kind` and
 97        # `accuracy`
 98        if self.stencil is not None:
 99            if not is_iterable(self.stencil):
100                raise TypeError(
101                    f"The value of the argument `stencil` ({self.stencil}) is not an iterable"
102                )
103
104            if not np.allclose(sorted(self.stencil), self.stencil):
105                raise ValueError(
106                    f"The value of the argument `stencil` ({self.stencil}) "
107                    "should be an increasing list of numbers"
108                )
109
110            return
111
112        allowed_kinds = ["center", "forward", "backward"]
113
114        if self.kind not in allowed_kinds:
115            raise ValueError(
116                f"Only the following derivative kinds are allowed: {allowed_kinds}"
117            )
118
119        if self.accuracy < 1:
120            raise ValueError("The accuracy must be at least first-order")
121
122        if self.kind == "center":
123            npoints = (2 * np.floor((self.order + 1) / 2) - 2 + self.accuracy) // 2
124            self.stencil = np.arange(-npoints, npoints + 1, 1)
125        elif self.kind == "forward":
126            npoints = self.accuracy + self.order
127            self.stencil = np.arange(0, npoints + 1)
128        else:
129            npoints = self.accuracy + self.order
130            self.stencil = np.arange(-npoints, 1)

133class FisherDerivative(ABC):
134    r"""
135    Abstract class for computing derivatives using finite differences.
136
137    Notes
138    -----
139    The user must at least implement the `signal` method to be able to
140    instantiate a subclass.
141    Furthermore, the `covariance` method, if implemented, needs to return an
142    array that is shape-compatible with the output of the `signal` method.
143    """
144
145    @abstractmethod
146    def signal(
147        self,
148        *args: tuple[str, float],
149        **kwargs,
150    ):
151        """
152        The method to implement for computing the signal.
153
154        Parameters
155        ----------
156        args
157            the name(s) of the parameter(s) and their respective value(s)
158
159        kwargs
160            any keyword arguments to be passed
161
162        Returns
163        -------
164        array-like of floats
165        """
166        return NotImplemented

168    def covariance(
169        self,
170        *args: tuple[str, float],
171        **kwargs,
172    ):
173        """
174        The (optional) method to implement for computing the covariance.
175
176        Parameters
177        ----------
178        args
179            the name(s) of the parameter(s) and their respective value(s)
180
181        kwargs
182            any keyword arguments to be passed
183
184        Returns
185        -------
186        array-like of floats with a square shape
187
188        Raises
189        ------
190        * `NotImplementedError` if the user has not explicitly overridden the
191        method
192        """
193        raise NotImplementedError(
194            "The `covariance` method must be implemented first in order to be used"
195        )
196
197    def __call__(
198        self,
199        method: str,
200        *args: D,
201        **kwargs,
202    ):
203        r"""
204        Evaluates the derivative.
205
206        Parameters
207        ----------
208        method
209            the method to use for computing the derivative. Can be either
210            'signal' or 'covariance'.
211
212        arg: D
213            the derivative
214
215        kwargs
216            any keyword arguments passed to the method
217        """
218        _validate_derivatives(*args)
219
220        weights_arr = [find_diff_weights(arg.stencil, order=arg.order) for arg in args]
221        points_arr = [arg.stencil for arg in args]
222
223        denominator = np.prod([arg.abs_step**arg.order for arg in args])
224
225        stencils = np.array(list(product(*points_arr)))
226        weights = np.array([np.prod(_) for _ in product(*weights_arr)])
227
228        # zero-out elements which are close to zero (they are non-zero due to
229        # floating-point math)
230        weights = np.array([_ if np.abs(_) > 1e-10 else 0 for _ in weights])
231
232        # remove them from the stencil
233        stencils = stencils[np.nonzero(weights)]
234        weights = weights[np.nonzero(weights)]
235
236        return np.array(
237            np.sum(
238                [
239                    getattr(self, method)(
240                        *[
241                            (arg.name, arg.value + arg.abs_step * p)
242                            for arg, p in zip(args, point)
243                        ],
244                        **kwargs,
245                    )
246                    * weight
247                    / denominator
248                    for point, weight in zip(stencils, weights)
249                ],
250                axis=0,
251            )
252        )

254    def fisher_matrix(
255        self,
256        *args: D,
257        constant_covariance: bool = True,
258        latex_names: Optional[Collection[str]] = None,
259        **kwargs,
260    ):
261        r"""
262        Computes the Fisher matrix, $\mathsf{F}$, using finite differences.
263
264        Parameters
265        ----------
266        args
267            the derivatives (see description of `D`) for which we want to
268            compute the derivatives
269
270        constant_covariance
271            whether or not to treat the covariance as constant (default: true)
272
273        latex_names
274            the LaTeX names of the parameters that will be passed to the
275            `FisherMatrix`
276
277        kwargs
278            any other keyword arguments that should be passed to `signal` and
279            `covariance`
280
281        Returns
282        -------
283        instance of `FisherMatrix` with corresponding names and fiducials
284
285        Notes
286        -----
287        The derivative order is automatically set to 1 for each parameter, and
288        is ignored if passed to `D`.
289        """
290        # first we attempt to compute the covariance; if that fails, it means
291        # it hasn't been implemented, so we fail fast and early
292        names = np.array([_.name for _ in args])
293        values = np.array([_.value for _ in args])
294
295        covariance_matrix = self.covariance(*zip(names, values))
296        inverse_covariance_matrix = np.linalg.inv(covariance_matrix)
297
298        covariance_shape = np.shape(inverse_covariance_matrix)
299
300        signal_derivative = {}
301
302        # TODO parallelize
303        for arg in args:
304            signal_derivative[arg.name] = self(
305                "signal",
306                D(
307                    name=arg.name,
308                    value=arg.value,
309                    abs_step=arg.abs_step,
310                    kind=arg.kind,
311                    accuracy=arg.accuracy,
312                    stencil=arg.stencil,
313                    **kwargs,
314                ),
315            )
316
317        covariance_derivative = {}
318
319        # TODO parallelize
320        for arg in args:
321            if not constant_covariance:
322                covariance_derivative[arg.name] = self(
323                    "covariance",
324                    D(
325                        name=arg.name,
326                        value=arg.value,
327                        abs_step=arg.abs_step,
328                        kind=arg.kind,
329                        accuracy=arg.accuracy,
330                        stencil=arg.stencil,
331                        **kwargs,
332                    ),
333                )
334            else:
335                covariance_derivative[arg.name] = np.zeros(covariance_shape)
336
337        fisher_matrix = np.zeros([len(args)] * 2)
338
339        for (i, arg1), (j, arg2) in product(enumerate(args), repeat=2):
340            fisher_matrix[i, j] = (
341                signal_derivative[arg1.name]
342                @ inverse_covariance_matrix
343                @ signal_derivative[arg2.name]
344                + np.trace(
345                    inverse_covariance_matrix
346                    @ covariance_derivative[arg1.name]
347                    @ inverse_covariance_matrix
348                    @ covariance_derivative[arg2.name]
349                )
350                / 2
351            )
352
353        return FisherMatrix(
354            fisher_matrix,
355            names=names,
356            fiducials=values,
357            latex_names=latex_names,
358        )

314                    **kwargs,
315                ),
316            )
317
318        covariance_derivative = {}
319
320        # TODO parallelize
321        for arg in args:
322            if not constant_covariance:
323                covariance_derivative[arg.name] = self(
324                    "covariance",
325                    D(
326                        name=arg.name,
327                        value=arg.value,
328                        abs_step=arg.abs_step,
329                        kind=arg.kind,
330                        accuracy=arg.accuracy,
331                        stencil=arg.stencil,
332                        **kwargs,
333                    ),
334                )
335            else:
336                covariance_derivative[arg.name] = np.zeros(covariance_shape)
337
338        fisher_matrix = np.zeros([len(args)] * 2)
339
340        for (i, arg1), (j, arg2) in product(enumerate(args), repeat=2):
341            fisher_matrix[i, j] = (
342                signal_derivative[arg1.name]
343                @ inverse_covariance_matrix
344                @ signal_derivative[arg2.name]
345                + np.trace(
346                    inverse_covariance_matrix
347                    @ covariance_derivative[arg1.name]
348                    @ inverse_covariance_matrix
349                    @ covariance_derivative[arg2.name]
350                )
351                / 2
352            )
353
354        return FisherMatrix(
355            fisher_matrix,
356            names=names,
357            fiducials=values,
358            latex_names=latex_names,
359        )

Computes the Fisher matrix, $\mathsf{F}$, using finite differences.

Parameters

args the derivatives (see description of D) for which we want to compute the derivatives

constant_covariance whether or not to treat the covariance as constant (default: true)

latex_names the LaTeX names of the parameters that will be passed to the FisherMatrix

kwargs any other keyword arguments that should be passed to signal and covariance

Returns

instance of FisherMatrix with corresponding names and fiducials

Notes

The derivative order is automatically set to 1 for each parameter, and is ignored if passed to D.
